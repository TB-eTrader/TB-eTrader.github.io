[文档]def init_logger(call_file_path: str) -> logging.Logger:
    """初始化一个日志记录器对象.

    :param call_file_path: 调用本函数的文件的绝对路径.
    :return: ``logging.Logger`` 对象
    """
    base_name = os.path.basename(call_file_path).split('.')[0]
    sub_pkg_path = os.path.dirname(call_file_path)
    sub_pkg_name = os.path.basename(sub_pkg_path)
    pkg_path = os.path.dirname(sub_pkg_path)

    log_file_path = os.path.join(pkg_path, 'logs', sub_pkg_name)
    if not os.path.exists(log_file_path):
        os.makedirs(log_file_path)

    formatter = logging.Formatter('%(asctime)s %(levelname)7s %(filename)8s line %(lineno)4d | %(message)s ',
                                  datefmt='%Y-%m-%d %H:%M:%S')

    logger_ = logging.getLogger(base_name)
    logger_.setLevel(logging.DEBUG)

    # info handler
    info_handler = RotatingFileHandler(os.path.join(log_file_path, f'{base_name}-info.log'), maxBytes=10 * 1024 * 1024,
                                       backupCount=5, encoding='utf-8')
    info_handler.setLevel(logging.INFO)
    info_handler.setFormatter(formatter)

    # debug handler
    debug_handler = RotatingFileHandler(os.path.join(log_file_path, f'{base_name}-debug.log'), maxBytes=10 * 1024 * 1024,
                                        backupCount=5, encoding='utf-8')
    debug_handler.setLevel(logging.DEBUG)
    debug_handler.setFormatter(formatter)

    logger_.addHandler(info_handler)
    logger_.addHandler(debug_handler)
    return logger_




[文档]def valid_date(date_: str) -> str:
    """验证日期字符串是否合法.

    :param date_: 待验证的字符串
    :raise ValueError: 日期格式不合法
    :return: 格式化后的日期字符串
    """
    try:
        date_str = datetime.strptime(date_, '%Y-%m-%d').strftime('%Y-%m-%d')
        return date_str
    except ValueError:
        raise ValueError(f'Invalid date: {date_}, format should be YYYY-mm-dd')



def _get_db_url(remote_control: bool) -> str:
    """获取数据库的连接信息.

    :param remote_control: 是否使用私有云数据库的外部接口
    :return: 数据库连接字符串
    """
    from DATABASE_CONF import local_prod_url, dev_url, prod_url
    if remote_control:
        return local_prod_url
    else:
        ip = socket.gethostbyname(socket.gethostname())
        if ip == '10.97.6.3':  # 私有云
            return prod_url
        else:
            return dev_url



[文档]def create_engine_and_tabs(remote_control: bool, tab_name_list: list) -> tuple:
    """创建数据库引擎, 并生成表对象.

    :param remote_control: 是否使用私有云数据库的外部接口
    :param tab_name_list: 需要创建的表名列表
    :return: 算法用到的表对象, 引擎
    """
    url = _get_db_url(remote_control)
    engine_ = create_engine(url)
    meta = MetaData()

    tabs = {k: Table(k, meta, autoload_with=engine_) for k in tab_name_list}
    return tabs, engine_




[文档]def smooth_data(y: list, *, method: str, **kwargs) -> list:
    """根据设置的不同的method对序列y进行平滑处理。支持的平滑方法如下：

    savgol：Savitzky-Golay filter

    ma：Moving Average方法

    对于每一种方法，还可以传入对应的关键词参数来定制平滑内容。

    """
    assert method in ('savgol', 'ma')

    if method == 'savgol':
        assert 'window_length' in kwargs
        assert 'polyorder' in kwargs
        y_hat = savgol_filter(y, **kwargs)
        return y_hat

    if method == 'ma':
        assert 'window_length' in kwargs

        center = kwargs.get('center', False)

        window = int(kwargs['window_length'])

        maed = pd.DataFrame(y).rolling(window, center=center).mean().values.reshape(-1)
        for i in range(len(maed)):
            if pd.isna(maed[i]):
                maed[i] = y[i]

        if not kwargs.get('triangular'):
            triangular = False
        else:
            triangular = kwargs['triangular']
        if triangular:
            # ref:https://plotly.com/python/smoothing/#:~:text=Savitzky%2DGolay-,Triangular%20Moving%20Average,-Another%20method%20for
            tri_smoothed = []
            for i in range(len(maed) - window):
                tri_smoothed.append(np.mean(maed[i:i + window]))
            tri_smoothed.extend(maed[i + 1:])  # noqa
            return tri_smoothed

        return maed
