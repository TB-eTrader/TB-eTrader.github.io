[文档]def post_process(values: list) -> list:
    """对预测的结果进行后处理.

    处理的流程共分三步:

    1. 将数据进行平滑处理

    2. 将数据置在[40, 650]之间

    3. 将数据进行截断

    :param values: 原始预测结果
    :return: ``list``, 处理后的预测结果
    """
    post_vals = []
    for v in values:
        if v < 40:
            post_vals.append(40)
        elif v > 650:
            post_vals.append(650)
        else:
            post_vals.append(v)
    return post_vals




[文档]def calc_indicators(df: pd.DataFrame) -> list:
    """计算派生指标"""
    df['conn'] = df['value_30'].map(lambda x: x if x > 0 else 0)  # 联络线出力
    df['all_power'] = df['conn'] + df['value_3'] + df['value_2']
    # 新能源出力占总出力的比例
    df['ratio_1'] = df['value_2'] / df['all_power']

    # 联络线占新能源出力的比例
    df['ratio_2'] = df['conn'] / df['value_2']

    return ['ratio_1', 'ratio_2']

[文档]    def post_process(self, ori_vals: list) -> list:
        """对预测的结果进行后处理.

        处理的流程共分三步:

        1. 将数据进行平滑处理

        2. 将数据置在[40, 650]之间

        3. 将数据进行截断

        :param ori_vals: 原始预测结果
        :return: ``list``, 处理后的预测结果
        """
        # 1
        fore_values = self.smooth_data(ori_vals)

        # 2
        fore_values = np.array(fore_values)
        fore_values[fore_values < 40] = 40
        fore_values[fore_values > 650] = 650
        fore_values = fore_values.tolist()

        # 3
        for i in range(len(fore_values)):
            if fore_values[i] < 200:
                fore_values[i] = 40

        return fore_values

    def _save_node(self, predict_values: list) -> None:
        """将预测结果保存到数据库."""
        pt_list = self.tab_dict['pt_prediction_node_price']
        pt_detail = self.tab_dict['pt_prediction_node_price_detail']

        model_id = self.model_dt[self.iter_id][self.market_id]
        now = datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S')
        forecast_start_date = self.base_date.strftime('%Y-%m-%d')

        # 先查看是否已经存在对应日期的预测价格数据
        with self.engine.connect() as conn:
            sql = (select(pt_list.c.id)
                               .where(pt_list.c.model_id == model_id)
                               .where(pt_list.c.market_id == self.market_id)
                               .where(pt_list.c.forecast_start_date == forecast_start_date)
                               .where(pt_list.c.node_id == self.iter_id))
            self.logger.debug(gen_raw_sql(sql, self.engine))
            ret = conn.execute(sql)

        rowcount = ret.rowcount
        try:
            with self.engine.begin() as conn:
                if rowcount != 0:
                    self.logger.warning("对应日期的预测价格数据已经存在, 结果入库为更新操作.")
                    id_ = ret.one()[0]
                    sql = (select(pt_detail.c.id)
                           .where(pt_detail.c.prediction_node_price_id == id_,
                                  pt_detail.c.date == self.predict_df.iloc[0]['date'])
                           .order_by(pt_detail.c.time))
                    ret = conn.execute(sql)
                    res = ret.fetchall()
                    for i in range(len(predict_values)):
                        conn.execute(update(pt_detail)
                                     .where(pt_detail.c.id == res[i][0])
                                     .values(price=predict_values[i]))
                else:
                    self.logger.warning("对应日期的预测价格数据不存在, 结果入库为插入操作.")
                    ret = conn.execute(insert(pt_list).values(**{
                        'node_id': self.iter_id,
                        'model_id': model_id,
                        'market_id': self.market_id,
                        'forecast_start_date': forecast_start_date,
                        'update_time': now
                    }))
                    insert_id = ret.lastrowid
                    ins_vals = []
                    for i in range(len(predict_values)):
                        ins_vals.append({
                            'prediction_node_price_id': insert_id,
                            'price': predict_values[i],
                            'date': self.predict_df.iloc[i]['date'],
                            'time': self.predict_df.iloc[i]['time']
                        })

                    conn.execute(insert(pt_detail).values(ins_vals))
                self.logger.info('写入数据库成功')
        except:
            self.logger.error(traceback.format_exc())
            self.logger.info('写入数据库失败')
