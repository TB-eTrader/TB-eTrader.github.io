class DmlModel(nn.Module):
    """
    :meta private:
    """
    def __init__(self, input_size):
        super(DmlModel, self).__init__()
        self.fc1 = nn.Linear(input_size, 1024)
        self.fc2 = nn.Linear(1024, 512)
        self.fc3 = nn.Linear(512, 256)
        self.fc4 = nn.Linear(256, 128)
        self.fc5 = nn.Linear(128, 1)
        self.relu = nn.ReLU()

    def forward(self, x):
        x = self.relu(self.fc1(x))
        x = self.relu(self.fc2(x))
        x = self.relu(self.fc3(x))
        x = self.relu(self.fc4(x))
        x = self.fc5(x)
        return x

[文档]def run_model(train_df: pd.DataFrame, pred_df: pd.DataFrame) -> pd.DataFrame:
    """训练模型，并预测结果。

    :param train_df: 训练集
    :param pred_df: 预测集
    :return: DataFrame，预测的结果。
    """
    df_x = train_df[ORI_COLUMNS]
    scaler_x = MinMaxScaler(feature_range=(0, 1))
    inv_x = scaler_x.fit_transform(df_x.values)

    df_y = train_df[['y_value']]
    scaler_y = MinMaxScaler(feature_range=(0, 1))
    inv_y = scaler_y.fit_transform(df_y.values)

    div_hour = inv_x.shape[0] - 12
    train_x, val_x = inv_x[:div_hour, :], inv_x[div_hour:, :]
    train_y, val_y = inv_y[:div_hour, :], inv_y[div_hour:, :]

    input_size = df_x.shape[1]
    model = DmlModel(input_size)
    criterion = nn.MSELoss()
    optimizer = optim.Adam(model.parameters())

    train_dataset = TensorDataset(torch.tensor(train_x, dtype=torch.float32),
                                  torch.tensor(train_y, dtype=torch.float32))
    train_loader = DataLoader(train_dataset, batch_size=BATCH_SIZE, shuffle=True)

    num_epochs = NUM_EPOCHS

    patience = 2
    early_stopping_counter = 0
    best_val_loss = float('inf')

    for i in range(NUM_LOOP):
        for epoch in range(num_epochs):
            model.train()
            for batch_x, batch_y in train_loader:
                optimizer.zero_grad()
                outputs = model(batch_x)
                loss = criterion(outputs, batch_y)
                loss.backward()
                optimizer.step()

            model.eval()
            with torch.no_grad():
                val_outputs = model(torch.tensor(val_x, dtype=torch.float32))
                val_loss = criterion(val_outputs, torch.tensor(val_y, dtype=torch.float32))
                logger.debug(
                    f'Loop [{i + 1}/{NUM_LOOP}], \
                    Epoch [{epoch + 1}/{num_epochs}], \
                    Validation Loss: {val_loss.item():.4f}')

                if val_loss < best_val_loss:
                    best_val_loss = val_loss
                    early_stopping_counter = 0
                else:
                    early_stopping_counter += 1

                if early_stopping_counter >= patience:
                    logger.debug("Early stopping triggered.")
                    break

        if early_stopping_counter >= patience:
            break

        # Resetting model parameters
        model.reset_parameters()

    val_pred = scaler_x.transform(pred_df[ORI_COLUMNS].values)
    val_pred_tensor = torch.tensor(val_pred, dtype=torch.float32)

    with torch.no_grad():
        model.eval()
        pred_y = model(val_pred_tensor)

    pred_y = pred_y.numpy()
    inv_y = scaler_y.inverse_transform(pred_y)
    inv_y = np.around(inv_y, 2)

    pred_df.applymap(lambda x: "%.2f" % x)
    pred_df.loc[:, 'y_value'] = inv_y
    # # 修正结果
    # rectify_result(pred_df)
    return pred_df




[文档]def infer_date(date: datetime.date) -> tuple:
    """根据 ``date`` 推断出训练数据的起止日期和用于预测的数据的起止日期。

    :param date: 基准日期，也是训练数据的结束日期
    :return: 推断的日期组成的元组。
    """
    # 训练数据的长度为3天
    train_end_date = date
    train_start_date = train_end_date - datetime.timedelta(days=2)

    pred_start_date = date + datetime.timedelta(days=1)
    pred_end_date = date + datetime.timedelta(days=3)
    return train_start_date, train_end_date, pred_start_date, pred_end_date

def _get_load(start_date: datetime.date, end_date: datetime.date) -> pd.DataFrame:
    """从数据库中读取省调负荷，新能源出力以及非市场化机组出力数据。

    :param start_date: 数据开始日期
    :param end_date: 数据结束日期
    :return: DataFrame。
    """
    pt_load = tab_dict['pt_load']
    sql = select(pt_load.c.date, pt_load.c.time, pt_load.c.value, pt_load.c.power_type_id).where(
        and_(
            pt_load.c.province_id == PROVINCE_ID,
            pt_load.c.market_id == 3,
            pt_load.c.date >= start_date.strftime('%Y-%m-%d'),
            pt_load.c.date <= end_date.strftime('%Y-%m-%d'),
            pt_load.c.power_type_id.in_([1, 2, 3])
        )
    ).order_by(pt_load.c.date, pt_load.c.time)
    df = pd.read_sql(sql, engine)
    set_datetime_col(df, False)

    # 省调负荷
    province_dispatch = df[df['power_type_id'] == 1].copy()
    province_dispatch.rename(columns={'value': 'province_dispatch_load_power'}, inplace=True)
    province_dispatch.drop(['power_type_id'], axis=1, inplace=True)
    province_dispatch.set_index('datetime', inplace=True)

    # 新能源出力
    new_energy_forecast = df[df['power_type_id'] == 2].copy()
    new_energy_forecast.rename(columns={'value': 'new_energy_load_forecast_power'}, inplace=True)
    new_energy_forecast.drop(['power_type_id'], axis=1, inplace=True)
    new_energy_forecast.set_index('datetime', inplace=True)

    # 非市场化机组出力
    not_market_gene = df[df['power_type_id'] == 3].copy()
    not_market_gene.rename(columns={'value': 'not_market_gene_power_value'}, inplace=True)
    not_market_gene.drop(['power_type_id'], axis=1, inplace=True)
    not_market_gene.set_index('datetime', inplace=True)

    df = pd.concat(
        [province_dispatch, new_energy_forecast, not_market_gene],
        axis=1)
    return df

[文档]def set_datetime_col(df: pd.DataFrame, set_index: bool = True) -> None:
    """将 ``df`` 中的 ``date`` 列和 ``time`` 列进行合并，生成新列 ``datetime``。

    :param df: 原始数据
    :param set_index: 是否将 ``datetime`` 置为索引
    :return: ``None``，直接在原始数据上进行修改。
    """
    df['time'] = df['time'].astype(str)
    df.loc[df['time'] == '24:00', 'date'] = df['date'] + datetime.timedelta(1)
    df.loc[df['time'] == '24:00', 'time'] = '00:00'
    df['date'] = df['date'].astype(str)
    df['datetime'] = pd.to_datetime(df['date'] + ' ' + df['time'] + ':00')
    df.drop(['date', 'time'], axis=1, inplace=True)
    if set_index:
        df.set_index('datetime', inplace=True)




[文档]def calc_th_ratio(df: pd.DataFrame) -> None:
    """计算 ``th_ratio``。"""
    need_float_column = ['province_dispatch_load_power', 'exported_power', 'grid_capacity',
                         'new_energy_load_forecast_power', 'not_market_gene_power_value']
    for col in need_float_column:
        df[col] = df[col].astype(float)

    df['th_ratio'] = (df['province_dispatch_load_power'] + df['exported_power'] +
                      df['new_energy_load_forecast_power'] - df['not_market_gene_power_value']
                      ) / df['grid_capacity']

    if 'y_value' not in df:
        df['y_value'] = 0.0

def _rectify_detail(df_data, rec_name: str = 'minimal', rec_flag: int = 0):
    ignore_key = ['edge_value', 'yes_flag']
    rec_set = RECTIFY_ADJUST[rec_name]
    rec_value = rec_set['edge_value']
    yes_flag = rec_set['yes_flag']
    val_edge = RECTIFY_VAL_EDGE[rec_flag]

    if rec_flag == 0:
        df_data.loc[df_data['y_value'] < rec_value, 'y_value'] = rec_value
        if yes_flag:
            df_data.loc[df_data['th_ratio'] <= val_edge, 'y_value'] = rec_value
    elif rec_flag == 1:
        df_data.loc[df_data['y_value'] > rec_value, 'y_value'] = rec_value
        if yes_flag:
            df_data.loc[df_data['th_ratio'] >= val_edge, 'y_value'] = rec_value

    for key in rec_set.keys():
        if key in ignore_key:
            continue

        ref_set = rec_set[key]
        if not ref_set['on']:
            continue

        edge_val = ref_set['value']
        edge_sign = ref_set['sign']

        if edge_sign == 0:
            df_data.loc[df_data[key] < edge_val, 'y_value'] = rec_value
        elif edge_sign == 1:
            df_data.loc[df_data[key] <= edge_val, 'y_value'] = rec_value
        elif edge_sign == 2:
            df_data.loc[df_data[key] == edge_val, 'y_value'] = rec_value
        elif edge_sign == 3:
            df_data.loc[df_data[key] > edge_val, 'y_value'] = rec_value
        elif edge_sign == 4:
            df_data.loc[df_data[key] >= edge_val, 'y_value'] = rec_value



[文档]def get_forecast_cap() -> dict:
    """本算法的执行需要依赖一个非常重要的参数，即：全网开机容量。这个参数需要进行预估，
    目前是人工每天更新，然后存储在服务器中。本函数的作用就是从数据库中获取该参数。
    """
    # 逐日获取每一个预测日的最新的预测开机容量
    pred_cap = {}
    for date_ in pd.date_range(pred_start_date, pred_end_date):
        date_str = date_.strftime('%Y-%m-%d')
        sql = f"select value from pt_hd_on_cap where date='{date_str}' order by update_date desc limit 1"
        try:
            with engine.connect() as conn:
                value = conn.execute(text(sql)).fetchone()
            pred_cap[date_str] = value[0]
        except:
            logger.warning(f'获取{date_str}的开机容量失败')
            pred_cap[date_str] = DEFAULT_CAP
            continue
    return pred_cap
