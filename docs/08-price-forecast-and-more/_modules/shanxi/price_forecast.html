"""
.. mod-doc::

    现货电价按照交易类型可以分为 **日前电价** 和 **实时电价** 两类，按照适用结算主体又可以分为
    **节点电价** （用于发电侧结算）和 **统一出清价格** （用于售电侧结算）两类。
    上述组合一共有四种，即：**日前节点电价**、**日前统一出清价格**、**实时节点价格** 和 **实时统一出清价格**。

    本算法对上述四种电价进行预测。

    对于周期性的定时调度任务而言，每天21点至22点之间程序会预测日前价格；每天11点至12点之间，程序会预测实时价格。
    如果指定了 ``-C`` 参数，则预测的是统一出清价格；否则预测节点价格。在预测节点价格时，
    目前只预测特新场站和晖源场站所在的两个节点。

    一些使用示例如下：

    .. code-block:: bash

        # 打印帮助文档
        python -m shanxi.price_forecast -h
        # 强制预测基准日期为2024-6-1的日前统一出清价格
        python -m shanxi.price_forecast -C -F --market 3 -D 2024-6-1
        # 预测节点电价，基准日期和市场信息默认进行推断
        python -m shanxi.price_forecast -R

"""

[文档]def fill_weather_data(df: pd.DataFrame):
    """分时天气预测数据可能会存在缺失的记录，此函数的功能是对其进行简单填充"""
    index = pd.date_range('2022-1-1', '2022-1-2', freq='1H').map(lambda i: i.strftime('%H:00'))[:-1].tolist()
    fill_value = None

    # 首先判断第一个索引处是否有值,如果没有,则将后续的第一个值填充至第一个索引处
    try:
        _ = df.loc[index[0]]
    except KeyError:
        fill_value = df.iloc[0].values

    fill_values = []
    fill_indexes = []
    i = 0
    while i < len(index):
        idx = index[i]
        try:
            fill_value = df.loc[idx].values
            i += 1
        except KeyError:
            fill_values.append(fill_value)
            fill_indexes.append(idx)
            if i == len(index) - 1:
                break

            for j in range(i + 1, len(index)):
                try:
                    _ = df.loc[index[j]].values
                    i = j
                    break
                except KeyError:
                    i = j
                    if j != len(index) - 1:
                        fill_values.append(fill_value)
                        fill_indexes.append(index[j])
    return pd.DataFrame(data=fill_values, index=fill_indexes, columns=df.columns), index



def _handle_weather_data1(df: pd.DataFrame) -> pd.DataFrame:
    """对一天的实测分时天气数据进行特征化。"""
    # 为了能将各区县按照三个区域进行区分,先梳理每个区域对应的曲线的id列表
    zh_2_ids = {
        '晋北': [
            101100208, 101100207, 101100206, 101100904, 101100906,
            101100903, 101101005, 101101013, 101101008, 101101003],
        '晋中': [
            101101103, 101101109, 101101111, 101101110, 101100104,
            101100103, 101100102, 101100407, 101100404, 101100410,
            101100412, 101100302, 101100303],
        '晋南': [
            101100703, 101100710, 101100706, 101100714, 101100510,
            101100502, 101100509, 101100506, 101100805, 101100808,
            101100811, 101100813, 101100605, 101100604, 101100606, 101100603
        ]
    }
    df['windSpeed'] = df['windSpeed'].astype(float)
    columns = []
    dfs = []
    for area in zh_2_ids:
        if area == '晋北':
            columns.extend(['n_temp', 'n_wind_speed', 'n_cloud'])
        elif area == '晋中':
            columns.extend(['m_temp', 'm_wind_speed', 'm_cloud'])
        else:
            columns.extend(['s_temp', 's_wind_speed', 's_cloud'])

        area_df = df.query(f'cityId in {zh_2_ids[area]}')
        gpd = area_df.groupby(by=['time'])
        wea_df = gpd[['temp', 'windSpeed', 'cloud']].mean()
        filled_df, _ = fill_weather_data(wea_df)
        if not filled_df.empty:
            wea_df = pd.concat([filled_df, wea_df])
            pass
        dfs.append(wea_df)
    ret_df = pd.concat(dfs, axis=1)
    ret_df.columns = columns

    # 扩充成96点的数据
    df = pd.concat([ret_df] * 4).sort_index()
    index = pd.date_range('1920-1-1 00:15',
                          freq='15T', periods=96).map(lambda x: x.strftime('%H:%M')).tolist()
    index[-1] = '24:00'
    df['index'] = index
    df.set_index('index', inplace=True)
    return df

        # df.columns == ['date', 'time', 'in_load', 'out_load', 'in_ne', 'not_market']
        return df


    def _add_feature(self, df: pd.DataFrame,
                     for_train: bool = True,
                     price_type: str = 'da') -> pd.DataFrame:
        """为数据集增加特征。

        :param df: 数据集
        :param for_train: 是否为训练集
        :param price_type: 价格类型，日前（``da``）或实时（``rt``）
        :return: 增加特征后的数据集。
        """
        if for_train:
            # 上一天的价格与当天的时间戳对齐, 作为一个新特征
            df['one_day_shift_price'] = df['price'].shift(96)
            df = df.iloc[96:, :]

        logger.info(f'增加时间戳特征...')
        ts_fts = self._gen_ts_features(df)
        logger.info(f'时间戳特征增加完成')

        if price_type == 'da':
            if self.use_weather:
                logger.info(f'尝试增加天气特征...')
                try:
                    we_fts = ['n_temp', 'n_wind_speed', 'n_cloud', 'm_temp', 'm_wind_speed', 'm_cloud',
                              's_temp', 's_wind_speed', 's_cloud']
                    df = get_wea_data_and_feature(df, for_train)
                    logger.info('天气特征增加完成')
                except:
                    logger.warning('[失败]增加天气特征失败.')
                    self.use_weather = False
                    we_fts = []
            else:
                logger.warning(f'不使用天气特征.')
                we_fts = []
        else:
            # 实时电价预测过程中不使用天气特征
            we_fts = []

        # update the original feature list
        if for_train:
            self.fea_cols.extend(we_fts + ts_fts + ['one_day_shift_price'])
        else:
            self.fea_cols.extend(we_fts + ts_fts)
        logger.info('特征增加完成')
        return df


[文档]    def read_data(self, start_date: date, end_date: date, market_id: int, node_id: int = None, for_train: bool = True) -> None:
        """读取数据的函数。

        .. note::

            数据中可能存在空值，导致无法被利用。为了避免这种情况，读取数据的过程采用了逐日读取的方式。

            对于用于训练的数据而言，要求单日的所有基础数据都是完整的，如果存在缺失，则弃用该日的所有数据，
            并在一定范围内向前顺移，直至获取到足够天数的数据。

            对于用于预测的数据而言，要求单日的新能源出力数据是完整的（即在整个预测日期范围内都有数据），省调负荷和联络线可以缺失，
            算法会对缺失数据进行处理。

        .. todo::

            更新功能：预测日期范围内的新能源出力如果有缺失，则只跳过该日而预测那些数据完整的日期的价格。


        :param start_date: 数据的开始日期
        :param end_date: 数据的结束日期
        :param market_id: 日前or实时
        :param node_id: 节点的唯一标识，如果预测统一出清价格则无需指定
        :param for_train: 是否为训练数据，默认为 ``True``

        :raise ValueError: 如果数据缺失过多，则抛出异常

        :return: 无返回值，数据存储在类属性中。
        """
        days = (end_date - start_date).days + 1
        logger.debug(f'共需读取 {days} 天数据')
        if for_train:
            logger.info("逐日获取数据...")
            x = []
            shift_days = 0
            target_date = end_date
            y = []
            for i in range(days):
                while True:
                    if shift_days > 7:
                        err_msg = f'用于训练的数据缺失超过7天, 无法继续训练'
                        logger.error(err_msg)
                        raise ValueError(err_msg)
                    try:
                        ret = self.read_oneday_load(target_date)
                    except ValueError as e:
                        logger.error(f"读取数据错误: {e}")
                        target_date -= timedelta(1)
                        shift_days += 1
                        logger.warning('尝试获取前一天的数据')
                        continue
                    else:
                        logger.debug(f"读取 {target_date} 日的省调负荷,新能源出力与联络线数据完成")
                        try:
                            ret2 = self.read_oneday_price(target_date, market_id, node_id)
                        except ValueError as e:
                            logger.error(f"读取价格数据错误: {e}")
                            target_date -= timedelta(1)
                            shift_days += 1
                            logger.warning('尝试获取前一天的价格数据')
                            continue
                        else:
                            logger.debug(f"读取 {target_date} 日的价格数据完成")
                            x.append(ret)
                            y.append(ret2)  # noqa
                            target_date -= timedelta(1)
                            break
            self.train_df = pd.concat(x, ignore_index=True)
            self.train_df['ratio1'] = self.train_df['in_ne'] / (self.train_df['in_ne'] +
                                                                self.train_df['in_load'] +
                                                                self.train_df['out_load'] +
                                                                self.train_df['not_market'])
            self.train_df['ratio2'] = self.train_df['out_load'] / self.train_df['in_ne']
            y = pd.concat(y, ignore_index=True)
            self.train_df['price'] = y['price']
        else:
            try:
                x_ne = self._read_days_ne(start_date, end_date)
            except ValueError:
                raise ValueError('读取用于预测的新能源出力数据失败')
            x_ne.rename(columns={'value': 'in_ne'}, inplace=True)

            x_ins = []
            x_outs = []
            target_date = start_date
            for i in range(days):
                while True:
                    try:
                        df_in_load = self._read_oneday_in_load(target_date, allow_empty_df=True)
                        df_conn_load = self._read_oneday_conn_load(target_date, allow_empty_df=True)
                    except Exception as e:
                        logger.error(f"读取 {target_date} 日的省调负荷与联络线数据失败: {e}")
                        raise Exception(f"读取 {target_date} 日的省调负荷与联络线数据失败: {e}")
                    else:
                        logger.debug(f"读取省调负荷与联络线数据成功: {target_date}")
                        if df_in_load.empty:
                            if i == 0:
                                raise ValueError(f'{target_date} 日的省调负荷数据为空, 且无历史数据可以使用')
                            last_df = x_ins[-1].copy()
                            last_df['date'] = target_date
                            x_ins.append(last_df)
                        else:
                            x_ins.append(df_in_load)
                        if df_conn_load.empty:
                            if i == 0:
                                raise ValueError(f'{target_date} 日的联络线数据为空, 且无历史数据可以使用')
                            last_df = x_outs[-1].copy()
                            last_df['date'] = target_date
                            x_outs.append(last_df)
                        else:
                            x_outs.append(df_conn_load)
                        target_date += timedelta(1)
                        break

            x_in = pd.concat(x_ins, ignore_index=True)
            x_out = pd.concat(x_outs, ignore_index=True)
            x_out.rename(columns={'value': 'out_load'}, inplace=True)

            x_in_out = x_in.merge(x_out, on=['date', 'time'])
            self.predict_df = x_ne.merge(x_in_out, on=['date', 'time'])
            self.predict_df['ratio1'] = self.predict_df['in_ne'] / (self.predict_df['in_ne'] +
                                                                    self.predict_df['in_load'] +
                                                                    self.predict_df['out_load'] +
                                                                    self.predict_df['not_market'])
            self.predict_df['ratio2'] = self.predict_df['out_load'] / self.predict_df['in_ne']

        self.fea_cols = ['in_load', 'out_load', 'in_ne', 'not_market', 'ratio1', 'ratio2']  # original features


    @staticmethod
    def _gen_ts_features(df: pd.DataFrame) -> list:
        """基于数据集的时间和日期来生成新特征。

        :param df: 数据集
        :return: 新增的特征的名称组成的列表，新特征数据将直接在原始数据集上改写。
        """
        is_peak = []  # 是否为用电高峰
        hours = []  # 当前分段的小时
        weekdays = []  # 当前日期是星期几
        months = []  # 当前日期是哪个月
        is_weekend = []  # 是否为周末
        for i in range(len(df)):
            hour = int(df['time'].iloc[i].split(":")[0])
            date_ = df['date'].iloc[i]
            if 10 <= hour <= 15:
                is_peak.append(1)
            else:
                is_peak.append(0)
            wd = date_.weekday()
            weekdays.append(wd)
            if wd in [5, 6]:
                is_weekend.append(1)
            else:
                is_weekend.append(0)
            hours.append(hour)
            months.append(date_.month)

        df['is_peak'] = is_peak
        df['is_weekend'] = is_weekend
        df['hour_no'] = hours
        df['weekday_no'] = weekdays
        df['month_no'] = months
        return ['is_peak', 'is_weekend', 'hour_no', 'weekday_no', 'month_no']

[文档]    def predict(self,
                market_id: int,
                node_id: int = None,
                do_smooth: bool = True,
                price_type: str = 'da'
                ) -> list:
        """读取预测数据，进行特征工程，并进行预测。"""
        logger.info('读取预测数据...')
        self.read_data(self.pred_start_date, self.pred_end_date, market_id, node_id, for_train=False)

        logger.info('增加特征...')
        self.predict_df = self._add_feature(self.predict_df, for_train=False, price_type=price_type)

        logger.info('开始预测...')
        predict_days = int(self.predict_df.shape[0] / 96)
        one_day_ahead_price = self.train_df.iloc[-96:]['price'].to_list()
        pred_values = []

        # 由于会用到上一日的价格，因此需要逐天预测
        for i in range(predict_days):
            sub_df = self.predict_df.iloc[i * 96: (i + 1) * 96]
            sub_df.loc[:, 'one_day_shift_price'] = one_day_ahead_price
            X = sub_df[self.fea_cols + ['one_day_shift_price']]  # noqa
            y_pred = self.model.predict(X)
            y_pred = self._post_process(y_pred)
            if do_smooth:
                y_pred = smooth_data(y_pred, method='ma', window_length=3, triangular=True, center=True)
            one_day_ahead_price = y_pred
            pred_values.extend(y_pred)
        return pred_values


    @staticmethod
    def _post_process(pred_vales: list) -> list:
        """对于预测的价格进行后处理。"""
        ret = []
        for i in pred_vales:
            if i < 200:
                ret.append(0)
            elif i > 1500:
                ret.append(1500)
            else:
                ret.append(i)
        return ret
