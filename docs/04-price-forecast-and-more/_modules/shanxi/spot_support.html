def _get_forecast_price_in_order(node_id: int, date_str: str, market_id: int) -> np.ndarray:
    """获取指定节点在指定日期的预测电价，包括日前电价和实时电价。

    获取过程是按照优先级从高到低的顺序进行的，优先级如下：

    1. 从推荐模型表中获取最近一段时间效果最好的模型对应的结果
    2. 从常规的节点电价预测表中获取

    :param node_id: 节点id
    :param date_str: 日期字符串，格式为 ``YYYY-MM-DD``
    :param market_id: 市场id，可选值为 ``{1, 3}``，分别表示实时市场和日前市场
    :raise ValueError: 如果获取到的电价数据为空
    :return: 电价数组。
    """
    # =====================================
    # tab1,tab2,tab3为推荐模型的相关表
    # =====================================
    tab1 = tab_dict['pt_prediction_node_price_recommend']
    tab2 = tab_dict['pt_prediction_node_price_recommend_type']
    tab3 = tab_dict['pt_prediction_node_price_recommend_type_detail']
    # =====================================
    # tab_ori1,tab_ori2为常规的节点电价预测表
    # =====================================
    tab_ori1 = tab_dict['pt_prediction_node_price']
    tab_ori2 = tab_dict['pt_prediction_node_price_detail']

    # 常规预测结果中, 不同节点的日前/实时电价预测模型的id
    ori_model_id = {
        1: {
            3: 16,
            1: 5
        },
        2: {
            3: 17,
            1: 6
        }
    }
    recommend_available = True
    with engine.connect() as conn:
        ret = conn.execute(select(tab1.c.id).where(tab1.c.node_id == node_id)
                           .where(tab1.c.market_id == market_id).where(tab1.c.date == date_str))
        best_1st_id = ret.fetchone()
        if best_1st_id is None:
            logger.warning(f'[最优结果]中没有对应记录(node_id={node_id},date={date_str},market_id={market_id})')
            recommend_available = False
        else:
            best_1st_id = best_1st_id[0]
            sql = (select(tab2.c.id)
                   .where(tab2.c.recommend_id == best_1st_id)
                   .order_by(tab2.c.last_seven_accuracy.desc()))

            ret = conn.execute(sql)
            rec_id = ret.fetchone()
            if rec_id is None:
                logger.warning(f'[最优结果]中没有对应记录(node_id={node_id},date={date_str},market_id={market_id})')
                recommend_available = False
            else:
                rec_id = rec_id[0]
                sql = select(tab3.c.prediction_price.label('price')).where(tab3.c.recommend_type_id == rec_id).order_by(tab3.c.time)
                ser = pd.read_sql(sql, engine)['price']
                if ser.isna().any():
                    logger.error(f'[最优结果]中节点{node_id}在{date_str}的`market_id={market_id}`电价预测数据不完整')
                    recommend_available = False
                else:
                    pass
        if not recommend_available:
            logger.warning('最优结果表中没有对应的预测数据，将从常规预测表中获取')
            # 在常规模型表中获取预测结果
            if market_id == 3:
                opt_forecast_dates = [
                    (pd.to_datetime(date_str) - pd.Timedelta(days=i)).strftime('%Y-%m-%d') for i in range(2, 6)
                ]
            else:
                opt_forecast_dates = [
                    (pd.to_datetime(date_str) - pd.Timedelta(days=i)).strftime('%Y-%m-%d') for i in range(3)
                ]
            id_ = None
            ser = None
            for forecast_start_date in opt_forecast_dates:
                sql = (select(tab_ori1.c.id)
                       .where(tab_ori1.c.node_id == node_id)
                       .where(tab_ori1.c.market_id == market_id)
                       .where(tab_ori1.c.model_id == ori_model_id[node_id][market_id])
                       .where(tab_ori1.c.forecast_start_date == forecast_start_date)
                       .order_by(tab_ori1.c.update_time.desc())
                       )
                ret = conn.execute(sql)
                id_ = ret.fetchall()
                if id_ is None or len(id_) == 0:
                    continue
                else:
                    sql = (select(tab_ori2.c.price)
                           .where(tab_ori2.c.date == date_str)
                           .where(tab_ori2.c.prediction_node_price_id == id_[0][0])
                           .order_by(tab_ori2.c.time)
                           )
                    df = pd.read_sql(sql, engine)
                    if df.empty:
                        continue
                    ser = df['price']
                    if ser.isna().any():
                        logger.warning(f'节点{node_id}在{date_str}的`market_id={market_id}`'
                                       f'电价预测数据不完整(forecast_start_date={forecast_start_date})')
                        continue
            if id_ is None or ser is None:
                logger.error(f'常规节点电价预测表中没有节点{node_id}在{date_str}的`market_id={market_id}`电价记录')
                raise ValueError(f'常规节点电价预测表中没有节点{node_id}在{date_str}的`market_id={market_id}`电价记录')
            logger.info('常规电价预测数据获取成功')
    return ser.values

[docs]def get_real_price(date_str: str, node_id: int) -> Dict[str, np.ndarray] | None:
    """
    获取指定节点在指定日期的真实的日前电价与实时电价。

    :param date_str: 日期字符串，格式为 ``YYYY-MM-DD``
    :param node_id: 节点id
    :return: ``pd.DataFrame`` 组成的字典，键为 ``'日前'`` 与 ``'实时'`` 或者 ``None``。
    """
    pt_node_price = tab_dict['pt_node_price']
    sql = (
        select(pt_node_price.c.ahead_day_price, pt_node_price.c.real_time_price)
        .where(pt_node_price.c.date == date_str).where(pt_node_price.c.node_id == node_id)
        .order_by(pt_node_price.c.time)
    )
    df = pd.read_sql(sql, engine)
    if df.empty:
        logger.error(f'节点{node_id}在{date_str}的电价数据不完整')
        return None

    if pd.to_numeric(df['ahead_day_price']).isna().any():
        logger.error(f'节点{node_id}在{date_str}的日前电价有空值')
        return None

    if pd.to_numeric(df['real_time_price']).isna().any():
        logger.error(f'节点{node_id}在{date_str}的实时电价有空值')
        return None

    arr1 = df['ahead_day_price'].values
    arr2 = df['real_time_price'].values
    return {
        '日前': arr1,
        '实时': arr2
    }




[docs]def acc(y_true, y_pred):
    """
    计算预测准确率。

    :param y_true: 真实值
    :param y_pred: 预测值
    :return: 准确率，取值范围为 ``[0, 1]``。
    """
    if not isinstance(y_true, np.ndarray):
        y_true = np.array(y_true)
    if not isinstance(y_pred, np.ndarray):
        y_pred = np.array(y_pred)

    y_true_without_0 = y_true[y_true != 0]
    y_pred_without_0 = y_pred[y_true != 0]

    sum_ = np.sum(np.abs((y_true_without_0 - y_pred_without_0) / y_true_without_0))
    acc_ = 1 - sum_ / len(y_true_without_0)
    return acc_ if acc_ > 0 else 0

[docs]    def calc_price_coffs(self, single_date: date, nid: int, days: int = 7):
        """获取从 ``single_date`` 起往前 ``days`` 天的真实电价（包含日前与实时）和预测电价（包含日前与实时）。
        并计算价差（日前价格减去实时价格）和预测准确率。

        :param single_date: 开始日期，也是要调整曲线的目标日期
        :param nid: 节点id
        :param days: 天数
        :return: ``None``。
        """
        end_date = single_date - pd.Timedelta(days=1)
        start_date = end_date - pd.Timedelta(days=days)

        logger.info(f'价差数据的日期范围: {start_date} ~ {end_date}')

        daily_da_acc = []
        daily_rt_acc = []
        df_diff = pd.DataFrame()

        for date_ in pd.date_range(start_date, end_date):
            date_ = date_.strftime('%Y-%m-%d')
            logger.debug(f'正在处理{date_}的电价差数据...')
            # 获取电价差并处理之
            real_pc_dict = get_real_price(date_, nid)
            fore_pc_dict = get_forecast_price(date_, nid)

            if real_pc_dict is None:
                continue

            real_da_pc = real_pc_dict['日前']
            real_rt_pc = real_pc_dict['实时']
            real_diff = real_da_pc - real_rt_pc
            _df_diff = pd.DataFrame({
                'hour': self.hour_int,
                'price_diff': real_diff
            })
            df_diff = pd.concat([df_diff, _df_diff], axis=0)

            fore_da_pc = fore_pc_dict['日前']
            fore_rt_pc = fore_pc_dict['实时']

            daily_da_acc.append(acc(real_da_pc, fore_da_pc))
            daily_rt_acc.append(acc(real_rt_pc, fore_rt_pc))

        df_diff = df_diff.groupby('hour').mean()
        self.his_price_diff = df_diff['price_diff'].values

        self.com_acc = np.average(daily_da_acc) * 0.5 + np.average(daily_rt_acc) * 0.5


    def __power_2_coe(self, power: np.ndarray) -> np.ndarray:
        """将调整后的功率转化为系数。"""
        coffs = np.divide(power, self.ori_power, out=np.zeros_like(power), where=self.ori_power != 0)
        return coffs

    def _save(self, coffs: np.ndarray):

            res = conn.execute(sql)
            count = res.one()[0]
            insert_values = []
            for i in range(len(coffs)):

                )

    def _build_model(self):
        """建立模型。"""
        self.model = pyo.ConcreteModel(name='曲线调整')
        self.model.num = pyo.RangeSet(1, 96)
        self.model.con_list = pyo.ConstraintList()

    def _get_data(self, date_str: str, sid: int):
        """获取用于曲线调整的原始数据，包括预测的价格和原始功率预测。

        :param date_str: 指定的日期
        :param sid: 场站id
        :return: ``None``，直接修改类属性。
        """
        price_dict = get_forecast_price(date_str, sid)
        da = price_dict['日前']
        rt = price_dict['实时']
        self.fore_price_diff = da - rt
        try:
            price_dict = get_real_price(date_str, sid)
            da = price_dict['日前']
            rt = price_dict['实时']
            self.real_price_diff = da - rt
        except:
            pass

        self.ori_power = get_ori_short_power(date_str, sid)

    def _add_constraints(self):
        """增加约束条件。"""
        cap = self.cap[self.station_id]

        # ============================================================
        # 新能源超额获利回收约束, 以小时为单位进行约束
        # ============================================================
        for i in range(24):
            start_idx = i * 4 + 1
            end_idx = (i + 1) * 4
            # 分时价差均价
            avg_price_diff = np.average([self.model.fore_price_diff[j] for j in range(start_idx, end_idx + 1)])
            # 分时出力之和
            avg_real_power = np.sum([self.model.ori_power[j] for j in range(start_idx, end_idx + 1)])
            # 分时调整后的申报功率之和
            sum_adj_power = 0
            for j in range(start_idx, end_idx + 1):
                sum_adj_power += self.model.x[j] * self.model.ori_power[j]

            # 相同时段的当前预测价差与历史价格的比率
            coe = np.abs(avg_price_diff) / np.abs(self.his_price_diff[i])
            if coe > 1:
                coe = 1

            if self.strategy_type == 'conservative':
                # 保守策略的额外处理
                # self.lambda_1 *= coe
                self.lambda_1 *= self.com_acc
                # self.lambda_2 *= coe
                self.lambda_2 *= self.com_acc

            if avg_price_diff > 0:  # 日前均价大于实时均价, 则添加上调约束
                # 根据实际出力占额定装机容量的比例来确定上调的幅度
                if avg_real_power > (val := self.THRESHOLD * cap):
                    self.model.con_list.add(expr=sum_adj_power <= avg_real_power * (1 + self.lambda_1))
                else:
                    self.model.con_list.add(expr=sum_adj_power <= val * (1 + self.lambda_1))
            elif avg_price_diff <= 0:  # 实时均价大于日前均价, 则添加下调约束
                self.model.con_list.add(expr=sum_adj_power >= avg_real_power * (1 - self.lambda_2))
            else:  # 不添加约束
                pass

        # ============================================================
        # 每个分段的申报电量不超过装机容量
        # ============================================================
        for i in range(1, 97):
            self.model.con_list.add(expr=self.model.x[i] * self.model.ori_power[i] <= cap)

        # ============================================================
        # 调整系数的增量不超过阈值, 以保证调整后的功率预测曲线的平滑性
        # ============================================================
        for i in range(2, 97):
            self.model.con_list.add(expr=(self.model.x[i] - self.model.x[i - 1]) <= self.smooth_para)
            self.model.con_list.add(expr=(self.model.x[i] - self.model.x[i - 1]) >= -self.smooth_para)

    def _post_process(self):
        """后处理。"""
        x_arr = np.asarray(self.x)
        adj_power = x_arr * self.ori_power
        smoothed_adj_power = np.asarray(smooth_data(adj_power, method='savgol', window_length=20, polyorder=3))
        # 将负值置为0
        smoothed_adj_power[smoothed_adj_power < 0] = 0
        # 原始功率为0的时段（光伏不出力），如果平滑后的值不为0，则置为0
        smoothed_adj_power[self.ori_power == 0] = 0
        return smoothed_adj_power

    def _run(self, date_str: str, station_id: int,
             strategy_type: str):
        """按照指定的策略类型对场站在指定日期的原始功率预测数据进行调整。

        :param date_str: 指定的日期
        :param station_id: 指定的场站id
        :param strategy_type: 指定的策略类型，可选值为：``{'conservative', 'radical'}``
        :return: ``None``，直接入库。
        """
        assert strategy_type in ('conservative', 'radical')
        st_zh = '保守策略' if strategy_type == 'conservative' else '激进策略'
        station_zh = '特新' if station_id == 1 else '晖源'
        logger.info(f'开始对{date_str}的{station_zh}场站执行{st_zh}曲线调整...')

        self.strategy_type = strategy_type
        self.station_id = station_id
        self.date_str = date_str

        # 1st, 建立模型
        self._build_model()

        # 2nd, 获取数据, 包括预测的电价数据和原始功率预测数据
        self._get_data(date_str, station_id)

        # 3rd, 定义变量与常量
        self.model.x = pyo.Var(self.model.num, domain=pyo.NonNegativeReals, initialize=1)
        self.model.ori_power = pyo.Param(self.model.num, initialize=lambda m, i: self.ori_power[i - 1])
        self.model.fore_price_diff = pyo.Param(self.model.num, initialize=lambda m, i: self.fore_price_diff[i - 1])

        # 4th, 定义目标函数
        self.model.OBJ = pyo.Objective(expr=sum(self.model.x[i] * self.model.ori_power[i] * self.model.fore_price_diff[i] for i in self.model.num),
                                       sense=pyo.maximize)

        # 5th, 添加约束
        self._add_constraints()

        solver = pyo.SolverFactory('glpk')
        results = solver.solve(self.model)
        # results = solver.solve(self.model, tee=True)
        # self.model.pprint()
        if results.solver.termination_condition == TerminationCondition.optimal:
            logger.info("发现最优解.")
            self.x = list(self.model.x.extract_values().values())
            post_power = self._post_process()
            # self.plot(post_power, date_str)
            ori_obj_value_under_real = round(sum(self.ori_power * self.real_price_diff), 2) if self.real_price_diff is not None else None
            ori_obj_value_under_fore = round(sum(self.ori_power * self.fore_price_diff), 2)
            post_obj_value_under_real = round(sum(post_power * self.real_price_diff), 2) if self.real_price_diff is not None else None
            post_obj_value_under_fore = round(sum(post_power * self.fore_price_diff), 2)
            # 打印以下内容:
            logger.debug('原始功率预测曲线在真实价差下的目标函数值:')
            logger.debug(f'\t{ori_obj_value_under_real}')
            logger.debug('调整后的功率预测曲线在真实价差下的目标函数值:')
            logger.debug(f'\t{post_obj_value_under_real}')
            logger.debug('原始功率预测曲线在预测价差下的目标函数值:')
            logger.debug(f'\t{ori_obj_value_under_fore}')
            logger.debug('调整后的功率预测曲线在预测价差下的目标函数值:')
            logger.debug(f'\t{post_obj_value_under_fore}')

            self._save(self.__power_2_coe(post_power))
            # self._save(self.x)
        elif results.solver.termination_condition == TerminationCondition.infeasible:
            logger.warning("问题不可行.")
        elif results.solver.termination_condition == TerminationCondition.unbounded:
            logger.warning("问题是无界的.")
        else:
            logger.warning("求解器未收敛到解决方案.")
