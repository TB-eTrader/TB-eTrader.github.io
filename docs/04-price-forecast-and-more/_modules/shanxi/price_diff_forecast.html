def convert_time_2_hour(time_str: str) -> str:
    if time_str.endswith('00'):
        return time_str
    else:
        hour = int(time_str.split(':')[0]) + 1
        if hour < 10:
            return f'0{hour}:00'
        else:
            return f'{hour}:00'



[docs]def get_price_diff(node_id: int, start_date: date, end_date: date):
    """获取指定节点在日期范围内的日前电价与实时电价的差值。

    :param node_id: 节点的id
    :param start_date: 开始日期
    :param end_date: 结束日期
    :return:
    """
    assert end_date >= start_date, f'结束日期应大于等于开始日期，{end_date} < {start_date}'
    tab = tab_dict['pt_node_price']
    sql = (select(tab.c.date, tab.c.time, tab.c.ahead_day_price, tab.c.real_time_price)
           .where(tab.c.date >= start_date.strftime('%Y-%m-%d'))
           .where(tab.c.date <= end_date.strftime('%Y-%m-%d'))
           .where(tab.c.node_id == node_id)
           .order_by(tab.c.date, tab.c.time))

    df = pd.read_sql(sql, engine)
    df['price_diff'] = df['ahead_day_price'] - df['real_time_price']
    if df['price_diff'].isnull().any():
        raise ValueError(f'节点{node_id}在{start_date}至{end_date}范围内数据存在空值')

    return df[['date', 'time', 'price_diff']]




[docs]def get_monthly_price_diff(node_id: int, month_str: str, return_df: bool = False):
    """获取指定节点在指定月份的日前电价与实时电价的差值。

    :param node_id: 节点的id
    :param month_str: 指定月份
    :param return_df: 是否返回DataFrame
    :return:
    """
    # 根据month_str获取该月的起止日期
    start_date = pd.to_datetime(month_str + '-01')
    end_date = start_date + pd.DateOffset(months=1) - pd.Timedelta(days=1)
    if return_df:
        return get_price_diff(node_id, start_date.date(), end_date.date())

    return get_price_diff(node_id, start_date.date(), end_date.date())['price_diff'].values




[docs]def plot_monthly_boxplot(node_id: int, start_month_str: str, end_month_str: str):
    """绘制指定节点在指定月份的日前电价与实时电价的差值的箱型图。

    :param node_id: 节点的id
    :param start_month_str: 开始月份
    :param end_month_str: 结束月份
    :return:
    """
    start_date = pd.to_datetime(start_month_str + '-01')
    end_date = pd.to_datetime(end_month_str + '-01')
    num_month = (end_date.year - start_date.year) * 12 + (end_date.month - start_date.month) + 1
    data_dt = {}
    for month in range(num_month):
        month_str = (start_date + pd.DateOffset(months=month)).strftime('%Y-%m')
        values = get_monthly_price_diff(node_id, month_str)
        data_dt[month_str[2:]] = values

    fig, ax = plt.subplots(figsize=(30, 5))
    # 不展示异常值
    ax.boxplot(data_dt.values(), showfliers=False)
    ax.set_xticklabels(data_dt.keys())
    # 旋转x轴文本
    plt.xticks(rotation=45)
    plt.show()



def plot_hourly_boxplot(node_id: int, start_month_str: str, end_month_str: str):
    start_date = pd.to_datetime(start_month_str + '-01')
    end_date = pd.to_datetime(end_month_str + '-01')
    num_month = (end_date.year - start_date.year) * 12 + (end_date.month - start_date.month) + 1
    hour_periods = ['01:00', '02:00', '03:00', '04:00', '05:00', '06:00', '07:00', '08:00', '09:00', '10:00', '11:00',
                    '12:00', '13:00', '14:00', '15:00', '16:00', '17:00', '18:00', '19:00', '20:00', '21:00', '22:00',
                    '23:00', '24:00']
    data_dt = {hour: [] for hour in hour_periods}
    for month in range(num_month):
        month_str = (start_date + pd.DateOffset(months=month)).strftime('%Y-%m')
        df = get_monthly_price_diff(node_id, month_str, True)
        df['hour'] = df['time'].apply(convert_time_2_hour)
        for hour in hour_periods:
            data_dt[hour].extend(df[df['hour'] == hour]['price_diff'].values.tolist())

    fig, ax = plt.subplots(figsize=(15, 5))
    # 不展示异常值
    ax.boxplot(data_dt.values(), showfliers=False)
    ax.set_xticklabels(data_dt.keys())
    ax.set_title(f'节点{node_id}在{start_month_str}至{end_month_str}月份的日前电价与实时电价的差值的箱型图')
    # 设置y轴的范围
    ax.set_ylim([-600, 500])
    # 旋转x轴文本
    plt.xticks(rotation=45)
    plt.show()

    def run_as_classification(self, base_date: date, eval_: bool = False):
        train_start_date, train_end_date, pred_start_date, pred_end_date = self.infer_date(base_date, pred_data_len=1)

        train_df = self.data_loader.get_load(train_start_date, train_end_date, power_type_ids=[1, 2, 3])
        train_pc = self.data_loader.get_price(train_start_date, train_end_date, is_clear=False, node_id=self.node_id,
                                              market_id=0)
        train_pc['diff'] = train_pc['da_price'] - train_pc['rt_price']
        train_pc['y'] = np.where(train_pc['diff'] > 0, 1, 0)
        fea_cols = [i for i in train_df.columns if i not in ['date', 'time']]

        model = HistGradientBoostingClassifier()
        model.fit(train_df[fea_cols], train_pc['y'])

        pred_df = self.data_loader.get_load(pred_start_date, pred_end_date, power_type_ids=[1, 2, 3])
        if eval_:
            pred_pc = self.data_loader.get_price(pred_start_date, pred_end_date, is_clear=False, node_id=self.node_id,
                                                 market_id=0)
            pred_pc['real_diff'] = pred_pc['da_price'] - pred_pc['rt_price']
            pred_pc['real_y'] = np.where(pred_pc['real_diff'] > 0, 1, 0)
            pred_df['pred_y'] = model.predict(pred_df[fea_cols])

            # 统计预测价差与真实价差同号的个数
            same_sign_count = 0
            for i in range(len(pred_df)):
                if pred_pc['real_y'].iloc[i] == pred_df['pred_y'].iloc[i]:
                    same_sign_count += 1

            # 统计光伏出力时段的预测价差与真实价差同号的个数
            same_sign_count_pv = 0
            den = 0
            for i in range(len(pred_df)):
                time_ = pred_df['time'].iloc[i]
                if '07:00' <= time_ <= '19:00':
                    if pred_pc['real_y'].iloc[i] == pred_df['pred_y'].iloc[i]:
                        same_sign_count_pv += 1
                    den += 1
            return same_sign_count, same_sign_count_pv

    def infer_date(target_date: date, train_data_len: int = 7, pred_data_len: int = 1):
        """计算日期。

        :param target_date: 目标日期，即需要预测价差的日期
        :param train_data_len: 训练集数据的天数
        :param pred_data_len: 预测集数据的天数
        :return:
        """
        train_end_date = target_date - datetime.timedelta(days=1)
        train_start_date = train_end_date - datetime.timedelta(days=train_data_len - 1)
        pred_start_date = target_date
        pred_end_date = pred_start_date + datetime.timedelta(days=pred_data_len - 1)
        return train_start_date, train_end_date, pred_start_date, pred_end_date

                       .where(tab.c.date >= self.pred_start_date)
                       .where(tab.c.date <= self.pred_end_date))
                conn.execute(sql)
                logger.info('删除完成')

            insert_vals = []
            dates = list(pd.date_range(self.pred_start_date, self.pred_end_date))
            for i in range(len(dates)):
                for j in range(len(times)):

    def eval_run(self, start_date: date, end_date: date):
        for d in pd.date_range(start_date, end_date):
