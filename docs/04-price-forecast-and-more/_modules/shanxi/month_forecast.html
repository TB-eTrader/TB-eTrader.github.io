def parse_args():
    description = "Monthly electricity price forecast algorithm for Shanxi province"
    parser = argparse.ArgumentParser(description=description, formatter_class=RawTextHelpFormatter)

    parser.add_argument('-R', '--remote_control', action='store_true',
                        help='Whether to use the private cloud database')
    parser.add_argument('--price_type', choices=('T', 'N'), default='T',
                        help='Specify the price type to be predicted, T for transaction price, N for node price.')
    parser.add_argument('-D', '--debug', action='store_true',
                        help='Whether to enable debug mode. If specified, '
                             'the private cloud database will not be used and `-R` will take no effect. '
                             'Also, the log will not be written to the file but printed to the console')
    parser.add_argument('-E', '--eval', action='store_true',
                        help='Whether to get the real price to evaluate the result.'
                             'Specified this parameter when validating the model')
    parser.add_argument('-N', '--node', action='store_true',
                        help='Node price will be predicted if specified, '
                             'otherwise the clearing price will be predicted.')
    parser.add_argument('-A', '--all', action='store_true',
                        help='Predict all nodes and clearing price when specified. '
                             'Thus the `-N` parameter will be ignored.')
    parser.add_argument('-M', '--month', nargs='?', type=valid_month,
                        help='Specify the month to be predicted, format: YYYY-mm')
    return parser.parse_args()



[docs]def get_date_range(year_month: str) -> Tuple[date, date]:
    """Given the month, return the start and end date of that month.

    :param year_month: The year and month, format: YYYY-mm
    :return: Start and end date of the ``year_month``.
    """
    year, month = map(int, year_month.split('-'))
    start_date = date(year, month, 1)
    if month == 12:
        end_date = date(year, 12, 31)
    else:
        end_date = date(year, month + 1, 1) - timedelta(days=1)
    return start_date, end_date




[docs]def get_base_date(start_date: date) -> date:
    """Return the base date based on ``start_date``.

    In this module, ``start_date`` is always the first day of the month to be predicted.

    Here is how to get base date: Starting from ``start_date`` and checking reversely
    to find the first day that new energy power is available, which is the base date.
    If no such day is found, raise an error.

    :param start_date: The first day of the month to be predicted
    :return: The base date.
    """
    counter = 1
    while True:
        if counter == 15:
            logger.error(f'No base date found, please check the data. start_date: {start_date}')
            raise ValueError(f'No base date found, please check the data. start_date: {start_date}')

        date_to_check = start_date - timedelta(days=counter)
        df = data_loader.get_load(date_to_check, date_to_check, power_type_ids=[2, 6, 7])
        if df.empty:
            counter += 1
            continue

        if df.shape[0] != 96:
            counter += 1
            continue

        if df.shape[1] != 5:
            counter += 1
            continue

        if df['value_2_3'].isna().any() or df['value_6_3'].isna().any() or df['value_7_3'].isna().any():
            counter += 1
            continue

        return date_to_check

[docs]def run_step1(train_df: pd.DataFrame, pred_df: pd.DataFrame) -> pd.DataFrame:
    """The deep learning part of step 2.

    :param train_df: The train dataset
    :param pred_df: The predict dataset
    :return: ``pd.DataFrame`` with the predicted price.
    """
    train_df = train_df.dropna()
    pred_df = pred_df.dropna()

    train_cols = [col for col in train_df.columns if col != 'y_value']

    df_x = train_df[train_cols]
    scaler_x = MinMaxScaler(feature_range=(0, 1))
    inv_x = scaler_x.fit_transform(df_x.values)

    df_y = train_df[['y_value']]
    scaler_y = MinMaxScaler(feature_range=(0, 1))
    inv_y = scaler_y.fit_transform(df_y.values)

    div_hour = inv_x.shape[0] - 12
    train_x, val_x = inv_x[:div_hour, :], inv_x[div_hour:, :]
    train_y, val_y = inv_y[:div_hour, :], inv_y[div_hour:, :]

    input_size = df_x.shape[1]
    model = DmlModel(input_size)
    criterion = nn.MSELoss()
    optimizer = optim.Adam(model.parameters())

    train_ds = TensorDataset(torch.tensor(train_x, dtype=torch.float32),
                             torch.tensor(train_y, dtype=torch.float32))
    train_loader = DataLoader(train_ds, batch_size=32, shuffle=True)

    num_epochs = 100
    patience = 2
    early_stopping_counter = 0
    base_val_loss = float('inf')

    for epoch in range(num_epochs):
        model.train()
        for batch_x, batch_y in train_loader:
            optimizer.zero_grad()
            outputs = model(batch_x)
            loss = criterion(outputs, batch_y)
            loss.backward()
            optimizer.step()

        model.eval()
        with torch.no_grad():
            val_outputs = model(torch.tensor(val_x, dtype=torch.float32))
            val_loss = criterion(val_outputs, torch.tensor(val_y, dtype=torch.float32))
            logger.info(f'  Epoch [{epoch + 1}/{num_epochs}], Val Loss: {val_loss.item():.4f}')
            if val_loss < base_val_loss:
                base_val_loss = val_loss
                early_stopping_counter = 0
            else:
                early_stopping_counter += 1
                if early_stopping_counter >= patience:
                    logger.info('  Early stopping triggered.')
                    break

    logger.info('  Train finished, predicting...')
    val_pred = scaler_x.transform(pred_df[train_cols].values)
    val_pred = torch.tensor(val_pred, dtype=torch.float32)

    with torch.no_grad():
        model.eval()
        pred_y = model(val_pred)

    pred_y = pred_y.numpy()
    inv_y = scaler_y.inverse_transform(pred_y)
    inv_y = np.around(inv_y, decimals=2)
    pred_df.applymap(lambda x: "%.2f" % x)
    pred_df['y_value'] = inv_y
    return pred_df




[docs]def run(start_date: date, end_date: date, node_id: int = None) -> None:
    """The entrance function of the algorithm.

    There are three steps in total:

    1. Train a sub-model using weather forecast data to predict the new energy power of specific dates.
    2. Train a sub-model using new energy power data to predict electricity price.
    3. Regard predicted new energy power data in step 1 as the input of the
        sub-model in step 2 and get the final results.

    :param start_date: The first day of the month to be predicted
    :param end_date: The last day of the month to be predicted
    :param node_id: The electricity node ID. If not specified, the clearing price will be predicted
    :return: ``None``. Results will be saved in to database.
    """
    global pred_ne_df_cache
    base_date = get_base_date(start_date)
    logger.info(f'The base date is: {base_date}')

    logger.info('Get the start and end date of train and predict dataset...')
    train_start_date = base_date - timedelta(days=365)
    train_end_date = base_date
    pred_start_date = base_date + timedelta(days=1)
    pred_end_date = end_date

    logger.info(f'##### step 1 #####')
    logger.info('Predict new energy power data using weather forecast data.')
    if pred_ne_df_cache is None:
        logger.info('Wind power part')
        logger.info('  [train part]Loading wind power and wind speed data...')
        train_x = data_loader.get_data_from_tj(train_start_date, train_end_date, 'w')
        train_y = data_loader.get_load(train_start_date, train_end_date, power_type_ids=[6])
        train_y.rename(columns={'value_6_3': 'y_value'}, inplace=True)
        agg_date_time(train_y)
        train_df = pd.concat([train_x, train_y], axis=1)
        logger.info('  [predict part]Loading wind power data...')
        pred_x = data_loader.get_data_from_tj(pred_start_date, pred_end_date, 'w')
        logger.info('  Train and predict...')
        pred_df_wind = run_step1(train_df, pred_x)
        logger.info('  Predict completed')

        logger.info('Solar power part')
        logger.info(f'  [train part]Loading pv power and irradiance data...')
        train_x = data_loader.get_data_from_tj(train_start_date, train_end_date, 'i')
        train_y = data_loader.get_load(train_start_date, train_end_date, power_type_ids=[7])
        train_y.rename(columns={'value_7_3': 'y_value'}, inplace=True)
        agg_date_time(train_y)
        train_df = pd.concat([train_x, train_y], axis=1)
        logger.info('  [predict part]Loading pv power data...')
        pred_x = data_loader.get_data_from_tj(pred_start_date, pred_end_date, 'i')
        logger.info('  Train and predict...')
        pred_df_pv = run_step1(train_df, pred_x)
        logger.info('  Predict completed')

        logger.info('Concatenate the results of two models...')
        power_res = pd.DataFrame()
        power_res.index = pred_df_wind.index
        power_res['load'] = pred_df_wind['y_value'] + pred_df_pv['y_value']
        pred_ne_df = power_res.groupby(power_res.index.hour)['load'].mean().reset_index()  # noqa
        pred_ne_df.drop(columns=['time'], inplace=True)
        logger.info('Concatenate completed')
        logger.info('New energy power data prediction finished.')
        pred_ne_df_cache = pred_ne_df.copy()
    else:
        logger.info('Load the cached results...')
        pred_ne_df = pred_ne_df_cache
        logger.info('Load completed')

    logger.info('##### step 2&3 #####')
    logger.info('Read real price and new energy data then train a model to predict price')
    month_num = 10
    start_month = calc_month(start_date.strftime('%Y-%m'), month_num)
    end_month = calc_month(start_date.strftime('%Y-%m'), 1)

    if PriceType == 'T':
        df_price = data_loader.get_hourly_transaction_price_m(start_month, end_month)
    else:
        df_price = data_loader.get_hourly_da_price(start_month, end_month, node_id=node_id)

    df_load = data_loader.get_hourly_ne_load(start_month, end_month)
    df = pd.concat([df_price, df_load], axis=1)
    pred_df = run_dml(df, pred_ne_df)
    logger.info('Price prediction finished')

    logger.info('Post processing...')
    post_process(pred_df, data_loader, Month)
    logger.info('Post processing finished')

    if Eval:
        logger.info('Get real price and evaluate the model...')
        # todo
        raise NotImplementedError
    else:
        logger.info('Saving results...')
        save_result(pred_df, start_date, end_date,
                    tab_dict, engine, ProvinceId,
                    PriceMap[node_id], MarketId, node_id)
        if PriceType == 'T':
            for _nid in NodeIds:
                if _nid != node_id:
                    save_result(pred_df, start_date, end_date,
                                tab_dict, engine, ProvinceId,
                                PriceMap[_nid], MarketId, _nid)

        logger.info('Results saved')
