[docs]def init_log(call_file_path: str, debug: bool = False):
    """初始化一个日志配置。

    :param call_file_path: 调用本函数的文件的绝对路径
    :param debug: 是否开启调试模式，默认为 ``False``，开启则将日志打印至控制台而非写入文件
    """
    base_name = os.path.basename(call_file_path).split('.')[0]
    sub_pkg_path = os.path.dirname(call_file_path)
    sub_pkg_name = os.path.basename(sub_pkg_path)
    pkg_path = os.path.dirname(sub_pkg_path)

    log_file_path = os.path.join(pkg_path, 'logs', sub_pkg_name)
    if not os.path.exists(log_file_path):
        os.makedirs(log_file_path)

    formatter = logging.Formatter('%(asctime)s %(levelname)7s %(filename)8s line %(lineno)4d | %(message)s ',
                                  datefmt='%Y-%m-%d %H:%M:%S')

    # info handler
    info_file_handler = RotatingFileHandler(os.path.join(log_file_path, f'{base_name}-info.log'), maxBytes=10 * 1024 * 1024,
                                            backupCount=5, encoding='utf-8')
    info_file_handler.setLevel(logging.INFO)
    info_file_handler.setFormatter(formatter)

    # debug handler
    debug_file_handler = RotatingFileHandler(os.path.join(log_file_path, f'{base_name}-debug.log'), maxBytes=10 * 1024 * 1024,
                                             backupCount=5, encoding='utf-8')
    debug_file_handler.setLevel(logging.DEBUG)
    debug_file_handler.setFormatter(formatter)

    # debug mode handler
    debug_stream_handler = logging.StreamHandler()
    debug_stream_handler.setLevel(logging.DEBUG)
    debug_stream_handler.setFormatter(formatter)

    if not debug:
        logging.basicConfig(
            handlers=[info_file_handler, debug_file_handler],
            level=logging.DEBUG
        )
    else:
        logging.basicConfig(
            level=logging.DEBUG,
        )

[docs]def _date_2_tdo(date_: date) -> str:
    """根据给定的日期，判断其是哪个旬，并以“年月旬”的形式返回。

    :param date_: 日期对象
    :return: 旬字符串

    .. testsetup::

        from utils import _date_2_tdo
        from datetime import date

    调用示例如下：

    >>> print(_date_2_tdo(date(2021, 1, 1)))
    2021-01-1
    >>> print(_date_2_tdo(date(2021, 1, 10)))
    2021-01-1
    >>> print(_date_2_tdo(date(2021, 1, 21)))
    2021-01-3

    """
    year, month, day = date_.year, date_.month, date_.day
    month_str = str(month) if month >= 10 else '0' + str(month)
    if day <= 10:
        return f'{year}-{month_str}-1'
    elif day <= 20:
        return f'{year}-{month_str}-2'
    else:
        return f'{year}-{month_str}-3'

[docs]def smooth_data(y: list, *, method: str, **kwargs) -> list:
    """根据设置的不同的method对序列y进行平滑处理。支持的平滑方法如下：

    savgol：Savitzky-Golay filter

    ma：Moving Average方法

    对于每一种方法，还可以传入对应的关键词参数来定制平滑内容。

    """
    assert method in ('savgol', 'ma')

    if method == 'savgol':
        assert 'window_length' in kwargs
        assert 'polyorder' in kwargs
        y_hat = savgol_filter(y, **kwargs)
        return y_hat

    if method == 'ma':
        assert 'window_length' in kwargs

        center = kwargs.get('center', False)

        window = int(kwargs['window_length'])

        maed = pd.DataFrame(y).rolling(window, center=center).mean().values.reshape(-1)
        for i in range(len(maed)):
            if pd.isna(maed[i]):
                maed[i] = y[i]

        if not kwargs.get('triangular'):
            triangular = False
        else:
            triangular = kwargs['triangular']
        if triangular:
            # ref:https://plotly.com/python/smoothing/#:~:text=Savitzky%2DGolay-,Triangular%20Moving%20Average,-Another%20method%20for
            tri_smoothed = []
            for i in range(len(maed) - window):
                tri_smoothed.append(np.mean(maed[i:i + window]))
            tri_smoothed.extend(maed[i + 1:])  # noqa
            return tri_smoothed

        return maed




[docs]def is_leap_year(year: int) -> bool:
    """判断一个年份是否是闰年。

    :param year: 待判断的年份
    :return: ``True``：闰年；``False``：平年。
    """
    # 闰年的条件是：年份能被4整除，但不能被100整除，或者能被400整除
    if (year % 4 == 0 and year % 100 != 0) or (year % 400 == 0):
        return True
    else:
        return False




[docs]def get_date_range(month: str, ten_days_order: int) -> tuple:
    """根据给出的月份及旬编号，确定旬的起止日期。

    .. example:

        >>> get_date_range('2024-1', 1)
        (datetime.date(2024, 1, 1), datetime.date(2024, 1, 10))

    :param month: 年月数据，年和月之间必须用 ``-`` 连接，如“2024-1”、“2023-04”等
    :param ten_days_order: 旬编号
    :return: 起止日期组成的数组。
    """
    a, b = month.split('-')
    year_int = int(a)
    month_int = int(b)

    if month_int in [1, 3, 5, 7, 8, 10, 12]:
        end = 31
    elif month_int in [4, 6, 9, 11]:
        end = 30
    else:  # month_int == 2
        if is_leap_year(year_int):
            end = 29
        else:
            end = 28

    if ten_days_order == 1:
        start_date = f'{a}-{b}-01'
        end_date = f'{a}-{b}-10'
    elif ten_days_order == 2:
        start_date = f'{a}-{b}-11'
        end_date = f'{a}-{b}-20'
    else:  # ten_days_order == 3
        start_date = f'{a}-{b}-21'
        end_date = f'{a}-{b}-{end}'
    return pd.to_datetime(start_date).date(), pd.to_datetime(end_date).date()

[docs]    def get_price(self, start_date: date, end_date: date,
                  is_clear: bool = True, market_id: int = 3,
                  node_id=None, zone_id=None):
        """获取价格数据。

        可以获取包括日前市场与实时市场的统一出清价格和节点价格数据。

        :param start_date: 数据开始日期
        :param end_date: 数据结束日期
        :param is_clear: 是否为统一出清价格，默认为 ``True``，即为统一出清价格
        :param market_id: 日前（3）、实时（1）或者两者都获取（0），默认为3
        :param node_id: 节点id，在 ``is_clear=False`` 时生效且必须指定
        :param zone_id: 区域id，在 ``is_clear=True`` 且 ``self.province_id=2`` 时生效且必须指定
        :return:
        """
        assert market_id in (0, 1, 3), f'`market_id` 只能为 0、1 或 3'
        assert end_date >= start_date, f'`end_date` 必须大于等于 `start_date`'
        tab = self.tab_dict['pt_clearing_price'] if is_clear else self.tab_dict['pt_node_price']
        conditions = [tab.c.date >= start_date.strftime('%Y-%m-%d'), tab.c.date <= end_date.strftime('%Y-%m-%d')]
        columns = [tab.c.date, tab.c.time]

        if not is_clear:
            assert node_id is not None, f'当 `is_clear=False` 时，`node_id` 必须指定'
            assert self._is_valid_nid(node_id), f'`node_id` {node_id} 不属于该省份（{self.province_id}）'
            conditions.append(tab.c.node_id == node_id)
            if market_id == 3:
                columns.append(tab.c.ahead_day_price)
            elif market_id == 1:
                columns.append(tab.c.real_time_price)
            else:
                columns.append(tab.c.ahead_day_price)
                columns.append(tab.c.real_time_price)
        else:
            conditions.append(tab.c.province_id == self.province_id)
            columns.append(tab.c.value)
            if market_id == 0:
                columns.append(tab.c.market_id)
                conditions.append(tab.c.market_id.in_((1, 3)))
            else:
                conditions.append(tab.c.market_id == market_id)
            if self.province_id == 2:  # 甘肃省
                assert zone_id is not None, f'在获取甘肃的统一出清价格时，`zone_id` 必须指定'
                conditions.append(tab.c.zone_id == zone_id)

        sql = select(*columns).where(and_(*conditions)).order_by(tab.c.date, tab.c.time)
        df = pd.read_sql(sql, self.engine)

        if is_clear:
            if market_id != 0:
                col = 'da_price' if market_id == 3 else 'rt_price'
                df.rename(columns={'value': col}, inplace=True)
            else:
                df = df.pivot(index=['date', 'time'], columns='market_id', values='value').reset_index()
                df.rename(columns={1: 'rt_price', 3: 'da_price'}, inplace=True)
        else:
            df.rename(columns={'ahead_day_price': 'da_price', 'real_time_price': 'rt_price'}, inplace=True)
        return df



[docs]    def get_load(self, start_date: date, end_date: date,
                 market_id: int = 3, power_type_ids=None):
        """从 ``pt_load`` 表中获取负荷类的数据。

        几种常见的 ``power_type_id`` 及其对应的含义如下：

        - 1：省调负荷
        - 2：新能源出力
        - 3：非市场化机组出力
        - 4：水电出力
        - 6：风电出力
        - 7：光伏出力

        完整的 ``power_type_id`` 的解释详见 ``pt_power_type`` 表。

        获取后的数据将被转换为宽表，除去 ``date`` 和 ``time`` 列外，
        其余列的命名规则为：``value_<power_type_id>_<market_id>``。

        :param start_date: 数据开始日期
        :param end_date: 数据结束日期
        :param market_id: 日前（3）、实时（1）或者两者都获取（0），默认为3
        :param power_type_ids: power类型id组成的列表，默认情况下为省调负荷
        :return:
        """
        power_type_ids = [1] if power_type_ids is None else power_type_ids
        assert market_id in (0, 1, 3), f'`market_id` 只能为 0、1 或 3'
        assert end_date >= start_date, f'`end_date` 必须大于等于 `start_date`'
        tab = self.tab_dict['pt_load']

        conditions = [tab.c.date >= start_date.strftime('%Y-%m-%d'),
                      tab.c.date <= end_date.strftime('%Y-%m-%d'),
                      tab.c.province_id == self.province_id,
                      tab.c.power_type_id.in_(power_type_ids)]
        if market_id != 0:
            conditions.append(tab.c.market_id == market_id)

        columns = [tab.c.date, tab.c.time, tab.c.value, tab.c.market_id, tab.c.power_type_id]
        sql = select(*columns).where(and_(*conditions)).order_by(tab.c.date, tab.c.time)
        df = pd.read_sql(sql, self.engine)
        df['market_id'] = df['market_id'].astype(str)
        df['power_type_id'] = df['power_type_id'].astype(str)

        df = df.pivot(index=['date', 'time'], columns=['power_type_id', 'market_id'], values='value').reset_index()
        mul_idx = df.columns
        new_cols = []
        for col in mul_idx:
            if col[1] == '':
                new_cols.append(col[0])
            else:
                new_cols.append('value_' + '_'.join(col))
        df.columns = new_cols
        return df

        """Calculate how many months between ``start_mont`` and ``end_month``."""
        start_month = datetime.strptime(start_month, '%Y-%m')

        tab = 'pt_tj_45d_1h_weather__partition_month'
        sql = f"""
        select fx_date, fx_time, {col}, area_code as county_code from {tab} 
        where fx_date between '{start_date}' and '{end_date}'
        and area_code in {tuple(county_codes)}
        """

        all_df['time'] = all_df['fx_date'].astype(str) + ' ' + all_df['fx_time'] + ':00'
        del all_df['fx_date'], all_df['fx_time']
        all_df['time'] = pd.to_datetime(all_df['time'])
        df = self._pivot_df(all_df, col)
        return df

        """获取分时的新能源出力数据。

        :param start_month: 开始月
        :param end_month: 结束月
        :param ten_days_order: 旬编号，如不指定，则默认是整月
        :return: 一个表示新能源出力的DataFrame，列名为：['value']。
        """

            assert ten_days_order in (1, 2, 3)

        # 确定数据的起止日期
        start_date = pd.to_datetime(start_month + '-01').date()
        end_year, end_month_ = end_month.split('-')
        _, end_day = calendar.monthrange(int(end_year), int(end_month_))
        end_date = pd.to_datetime(end_month + '-' + str(end_day)).date()

        # 获取日期范围内的全部逐15分钟数据
        df = self.get_load(start_date, end_date, power_type_ids=[2])

        # 按小时进行聚合
        df['hour'] = df['time'].apply(_15min_2_hour)
        df.rename(columns={'value_2_3': 'value'}, inplace=True)
        df = df.groupby(['date', 'hour'])[['value']].mean().reset_index()

        if ten_days_order:
            # 选出目标旬的数据
            df['ten_days'] = df['date'].map(_date_2_tdo)
            df = df[df['ten_days'].map(lambda x: x.split('-')[2] == str(ten_days_order))]

            # 每个旬的每个小时，只保留一个值
            df = df.groupby(['ten_days', 'hour'])[['value']].mean().reset_index()
            df['month'] = df['ten_days'].map(lambda x: x[:7])
            df['index'] = df['month'] + '-' + df['hour']
            df.drop(columns=['month', 'hour', 'ten_days'], inplace=True)
            df.set_index('index', inplace=True)
        else:
            df['month'] = df['date'].map(lambda x: x.strftime('%Y-%m'))
            # 每个月的每个小时，只保留一个值
            df = df.groupby(['month', 'hour'])[['value']].mean().reset_index()
            df['index'] = df['month'] + '-' + df['hour']
            df.drop(columns=['month', 'hour'], inplace=True)
            df.set_index('index', inplace=True)

        return df

        gp['index'] = gp['month'] + '-' + gp['hour']
        gp.drop(columns=['month', 'hour'], inplace=True)
        gp.set_index('index', inplace=True)
