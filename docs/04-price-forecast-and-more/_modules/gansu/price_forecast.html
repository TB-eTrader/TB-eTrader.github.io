    def fill_weather_data(df: pd.DataFrame) -> pd.DataFrame | None:
        """For unknown reasons, some weather data is missing, this function is used
        to find missing values' indexes and fill them simply.

        The input dataframe should contain only one day's data, and `date` and `time`
        must in columns.
        """
        index = pd.date_range('2022-1-1', '2022-1-2', freq='1H').map(lambda i: i.strftime('%H:00'))[:-1].tolist()

        # first to see if `time='00:00'` is not empty
        if df[df['time'] == '00:00'].empty:
            # use the first row to fill
            fill_value = df.iloc[0].values.tolist()
        else:
            fill_value = []

        fill_values = []
        fill_indexes = []
        i = 0
        while i < len(index):
            idx = index[i]
            sub_df = df[df['time'] == idx]
            if not sub_df.empty:
                fill_value = sub_df.iloc[0].values.tolist()
                i += 1
            else:
                fill_values.append(fill_value)
                fill_indexes.append(idx)
                if i == len(index) - 1:
                    break
                for j in range(i + 1, len(index)):
                    if not df[df['time'] == index[j]].empty:
                        i = j
                        break
                    else:
                        i = j
                        if j != len(index) - 1:
                            fill_values.append(fill_value)
                            fill_indexes.append(index[j])
        if fill_values:
            fill_df = pd.DataFrame(fill_values, columns=df.columns)
            fill_df['time'] = fill_indexes
            df = pd.concat([df, fill_df]).sort_values(by=['time']).reset_index(drop=True)
            return df
        else:
            return None

    def handle_nan(df: pd.DataFrame) -> None:
        """处理nan值"""
        if df['value_'].isnull().sum() == len(df):
            raise Exception('The dataframe is empty.')

        dates = df['date'].value_counts().index.sort_values()
        for i, date in enumerate(dates):
            sub_df = df[df['date'] == date]
            if sub_df['value_'].isnull().sum() > 0:
                if i == 0:
                    for j in range(1, len(dates)):
                        if df[df['date'] == dates[j]]['value_'].isnull().sum() == 0:
                            df.loc[df['date'] == date, 'value_'] = df[df['date'] == dates[j]]['value_'].values
                            break
                        else:
                            continue
                    pass
                else:
                    df.loc[df['date'] == date, 'value_'] = df[df['date'] == dates[i - 1]]['value_'].values
            else:
                continue



[docs]    def get_oneday_price(self, single_day: datetime.date, iter_id: int | str, market_id: int = 3) -> pd.DataFrame:
        """获取指定日期的日前或实时价格数据.

        :param single_day: 日期
        :param iter_id: 节点id或区域标识
        :param market_id: 日前or实时
        :raise ValueError: 数据不完整或存在空值
        :return: DataFrame
        """
        if isinstance(iter_id, int):
            price_tab = self.tab_dict['pt_node_price']
            col = 'ahead_day_price' if market_id == 3 else 'real_time_price'
            sql = (
                select(price_tab.c[col].label("price"), price_tab.c.time, price_tab.c.date)
                .where(price_tab.c.node_id == iter_id, price_tab.c.date == single_day.strftime('%Y-%m-%d'))
                .order_by(price_tab.c.time)
            )
            self.logger.debug(gen_raw_sql(sql, self.engine))
        else:
            price_tab = self.tab_dict['pt_clearing_price']
            sql = (
                select(price_tab.c.value.label("price"), price_tab.c.time, price_tab.c.date)
                .where(price_tab.c.date == single_day.strftime('%Y-%m-%d'), price_tab.c.zone_id == self.zone_id[iter_id],
                       price_tab.c.market_id == market_id)
                .order_by(price_tab.c.time)
            )
        df = pd.read_sql(sql, self.engine)
        # 数据校验
        msg = None
        if df.empty:
            msg = f'{single_day} 日的价格数据为空'
        if df.shape[0] != 96:
            msg = f'{single_day} 日的价格数据不完整'
        if df.isnull().any().any():
            msg = f'{single_day} 日的价格数据存在空值'
        if msg:
            self.logger.error(self.indentation + msg)
            raise ValueError(msg)
        return df



[docs]    def get_1day_load(self, single_day: datetime.date, power_type_ids: Union[int, List[int]]) -> pd.DataFrame:
        """获取指定日期的甘肃省的日前省调负荷/新能源出力/水电出力数据.

        :param single_day: 日期
        :param power_type_ids: 电力类型id或id列表
        :raise ValueError: 数据不完整或存在空值
        :return: DataFrame
        """
        pt_load = self.tab_dict['pt_load']
        if isinstance(power_type_ids, int):
            power_type_ids = [power_type_ids]

        sql = (
            select(pt_load.c.value, pt_load.c.time, pt_load.c.date, pt_load.c.power_type_id)
            .where(pt_load.c.date == single_day.strftime('%Y-%m-%d'))
            .where(pt_load.c.province_id == self.province_id)
            .where(pt_load.c.market_id == 3)
            .where(pt_load.c.power_type_id.in_(power_type_ids))
            .order_by(pt_load.c.time)
        )
        self.logger.debug(self.indentation + 'sql:\n' + gen_raw_sql(sql, self.engine))
        df = pd.read_sql(sql, self.engine)
        if df.empty:
            self.logger.error(self.indentation + f'负荷数据({power_type_ids})为空, 日期:{single_day}')
            raise ValueError(f'负荷数据({power_type_ids})为空, 日期:{single_day}')
        if len(df) != len(power_type_ids) * 96:
            self.logger.error(self.indentation + f'负荷数据({power_type_ids})不完整, 日期:{single_day}')
            raise ValueError(f'负荷数据({power_type_ids})不完整, 日期:{single_day}')
        if df.isnull().any().any():
            self.logger.error(self.indentation + f'负荷数据({power_type_ids})存在空值, 日期:{single_day}')
            raise ValueError(f'负荷数据({power_type_ids})存在空值, 日期:{single_day}')
        pivot_df = pd.pivot_table(df, index=['date', 'time'], columns='power_type_id', values='value')
        pivot_df.columns = [f'value_{col}' for col in pivot_df.columns]
        return pivot_df.reset_index()

    def smooth_data(y: list) -> list:
        """Smooth the data."""
        maed = pd.DataFrame(y).rolling(3, center=True).mean().values.reshape(-1)
        for i in range(len(maed)):
            if pd.isna(maed[i]):
                maed[i] = y[i]

        # ref:https://plotly.com/python/smoothing/#:~:text=Savitzky%2DGolay-,Triangular%20Moving%20Average,-Another%20method%20for
        tri_smoothed = []
        for i in range(len(maed) - 3):
            tri_smoothed.append(np.mean(maed[i:i + 3]))
        tri_smoothed.extend(maed[i + 1:])  # noqa
        return tri_smoothed

[docs]    def add_feature(self, df: pd.DataFrame) -> Tuple[pd.DataFrame, list]:
        """添加特征"""
        if self.ori_fea_cols is None:
            self.ori_fea_cols = []
            cols = [col for col in df.columns if col.startswith('value_')]
            price = df['price'].values
            corrs = []
            for col in cols:
                corrs.append(np.abs(np.corrcoef(df[col].values, price)[0, 1]))
            idx = np.argsort(corrs)[::-1]
            for i in range(3):
                if corrs[idx[i]] > 0.6:
                    self.ori_fea_cols.append(cols[idx[i]])
            if not self.ori_fea_cols:
                self.logger.warning('价格与各类负荷的相关系数均不高, 只使用第一种负荷作为特征')
                self.ori_fea_cols = [cols[0]]
            self.logger.info(f'选择使用的特征为:{self.ori_fea_cols}')

        self.logger.info(f'增加时间戳特征...')
        ts_fts = self._gen_ts_features(df)

        return df, self.ori_fea_cols + ts_fts

[docs]    def read_data(self, start_date: datetime.date, end_date: datetime.date, for_train: bool,
                  iter_id: int | str, market_id: int) -> None:
        """逐日读取数据.

        .. note::

            逐日读取的目的是为了防止某一天的数据存在空值或者缺失等情况对模型训练/预测造成影响.
            如果是训练过程, 则从 ``end_date`` 开始向前读取; 如果是预测过程, 则从 ``start_date`` 开始向后读取.

        :param start_date: 数据开始日期
        :param end_date: 数据结束日期
        :param for_train: 是否为训练数据
        :param iter_id: 节点id或区域标识
        :param market_id: 日前or实时
        :return: ``None``
        """
        days = (end_date - start_date).days + 1
        msg = '训练' if for_train else '预测'
        self.logger.info(f'共需要读取{days}天的数据用于{msg}')
        if for_train:
            target_date = end_date
            load_df = pd.DataFrame()
            shift_steps = 0
            self.indentation = '  '
            for i in range(days):
                self.logger.info(f'{self.indentation}读取{target_date}日数据...')
                while True:
                    if shift_steps > 7:
                        self.logger.error(f"{self.indentation}数据滞后超过7天, 无法获取数据, 请检查数据源是否正常")
                        raise Exception("数据滞后超过7天, 无法获取数据, 请检查数据源是否正常")

                    ret = self.get_one_day_data_for_train(target_date, iter_id, market_id)
                    if isinstance(ret, bool):
                        self.logger.warning(f'{self.indentation}[{target_date}]获取数据失败, 将向前推移以获取数据')
                        target_date -= datetime.timedelta(1)
                        shift_steps += 1
                    else:
                        assert isinstance(ret, pd.DataFrame)
                        load_df = pd.concat([load_df, ret])
                        target_date -= datetime.timedelta(1)
                        self.logger.debug(f'{self.indentation}[{target_date}]数据获取成功')
                        break
            self.indentation = ''
            self.train_df = load_df
        else:  # 逐日获取数据, 如果某天的数据存在缺失, 则针对缺失数据的类型, 进行不同的处理
            # ========================================================
            #  获取训练集中最新的一天的数据, 以用于补充测试集的缺失数据.
            # ========================================================
            df_train_max = self.train_df[self.train_df['date'] == self.train_df['date'].max()].copy()

            self.logger.info(self.indentation + '逐日获取数据...')
            df1s = []
            df2s = []
            df3s = []
            df4s = []
            df30s = []
            self.indentation = '  '
            for day in pd.date_range(start_date, end_date):
                date = day.date()
                self.logger.info(f'{self.indentation}读取{date}日数据...')
                try:
                    self.logger.debug(self.indentation + f'[{date}]获取省调负荷数据...')
                    df1 = self.get_1day_load(date, 1)
                    self.logger.debug(self.indentation + f'成功')
                except:
                    self.logger.warning(self.indentation + f'[{date}][failed]获取省调负荷数据失败, 将用训练集中的数据进行补充')
                    df1 = df_train_max[['date', 'time', 'value_1']]
                    df1.loc[:, 'date'] = date

                try:
                    self.logger.debug(self.indentation + f'[{date}]获取非市场化机组出力数据...')
                    df3 = self.get_1day_load(date, 3)
                    self.logger.debug(self.indentation + f'成功')
                except:
                    self.logger.warning(self.indentation + f'[{date}][failed]获取非市场化机组出力数据失败, 将用训练集中的数据进行补充')
                    df3 = df_train_max[['date', 'time', 'value_3']]
                    df3.loc[:, 'date'] = date

                try:
                    self.logger.debug(self.indentation + f'[{date}]获取水电出力数据...')
                    df4 = self.get_1day_load(date, 4)
                    self.logger.debug(f'  成功')
                except:
                    self.logger.warning(self.indentation + f'[{date}][failed]获取水电出力数据失败, 将用训练集中的数据进行补充')
                    df4 = df_train_max[['date', 'time', 'value_4']]
                    df4.loc[:, 'date'] = date

                try:
                    self.logger.debug(self.indentation + f'[{date}]获取联络线数据...')
                    df30 = self.get_1day_conn(date)
                    self.logger.debug(self.indentation + f'成功')
                except:
                    self.logger.warning(self.indentation + f'[{date}][failed]获取联络线数据失败, 将用训练集中的数据进行补充')
                    df30 = df_train_max[['date', 'time', 'value_30']]
                    df30.loc[:, 'date'] = date

                try:
                    self.logger.debug(self.indentation + f'[{date}]获取新能源出力数据...')
                    df2 = self.get_1day_load(date, 2)
                    self.logger.debug(self.indentation + f'成功')
                except:
                    self.logger.warning(self.indentation + f'[{date}][failed]获取新能源出力数据失败, 将用天气预报数据进行预测')
                    try:
                        df2 = self.predict_new_energy(date)
                        self.logger.info(self.indentation + '[success!]利用天气预报预测新能源出力成功')
                    except:
                        self.logger.warning(traceback.format_exc())
                        self.logger.warning(self.indentation + f'[failed!]利用天气数据预测新能源出力失败, 将用训练集中的数据进行补充')
                        df2 = df_train_max[['date', 'time', 'value_2']]
                        df2.loc[:, 'date'] = date
                self.logger.info(f'[{date}]数据获取完成')
                df1s.append(df1)
                df2s.append(df2)
                df3s.append(df3)
                df4s.append(df4)
                df30s.append(df30)
            self.indentation = ''
            df1 = pd.concat(df1s)
            df2 = pd.concat(df2s)
            df3 = pd.concat(df3s)
            df4 = pd.concat(df4s)
            df30 = pd.concat(df30s)

            df = pd.merge(df1, df2, on=['date', 'time']).merge(df3, on=['date', 'time']).merge(df4, on=['date', 'time']).merge(df30, on=['date', 'time'])
            self.predict_df = df

    def _save_node(self, predict_values: list) -> None:
        """将预测结果保存到数据库."""
        pt_list = self.tab_dict['pt_prediction_node_price']
        pt_detail = self.tab_dict['pt_prediction_node_price_detail']

        model_id = self.model_dt[self.iter_id][self.market_id]
        now = datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S')
        forecast_start_date = self.base_date.strftime('%Y-%m-%d')

        # 先查看是否已经存在对应日期的预测价格数据
        with self.engine.connect() as conn:
            ret = conn.execute(select(pt_list.c.id)
                               .where(pt_list.c.model_id == model_id)
                               .where(pt_list.c.market_id == self.market_id)
                               .where(pt_list.c.forecast_start_date == forecast_start_date)
                               .where(pt_list.c.node_id == self.iter_id))
        rowcount = ret.rowcount
        try:
            with self.engine.begin() as conn:
                if rowcount != 0:
                    self.logger.warning("对应日期的预测价格数据已经存在, 先删除.")
                    id_ = ret.one()[0]
                    conn.execute(delete(pt_detail).where(pt_detail.c.prediction_node_price_id == id_))
                    conn.execute(delete(pt_list).where(pt_list.c.id == id_))

                ret = conn.execute(insert(pt_list).values(**{
                    'node_id': self.iter_id,
                    'model_id': model_id,
                    'market_id': self.market_id,
                    'forecast_start_date': forecast_start_date,
                    'update_time': now
                }))
                insert_id = ret.lastrowid
                ins_vals = []
                for i in range(len(predict_values)):
                    ins_vals.append({
                        'prediction_node_price_id': insert_id,
                        'price': predict_values[i],
                        'date': self.predict_df.iloc[i]['date'],
                        'time': self.predict_df.iloc[i]['time']
                    })

                conn.execute(insert(pt_detail).values(ins_vals))
            self.logger.info('写入数据库成功')
        except:
            self.logger.error(traceback.format_exc())
            self.logger.info('写入数据库失败')
