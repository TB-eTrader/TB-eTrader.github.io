def parse_args():
    parser = argparse.ArgumentParser(formatter_class=RawTextHelpFormatter)
    parser.add_argument('-P', '--prod_env', action='store_true', help='写入生产环境数据库')
    parser.add_argument('--node', nargs='?', default='123,124,129',
                        help='需要预测的节点id. 如果有多个节点, 则它们之间用英文逗号分割.')

    date_help_msg = ("执行预测的日期, 格式为: 'YYYY-mm-dd', 如果不指定, 则date=today."
                     "\n- 如果market=3, 则预测`date`+2单日的价格."
                     "\n- 如果market=1, 则预测`date`+1单日的价格.")
    parser.add_argument('-D', '--date', nargs='?', default=None, type=valid_date,
                        help=date_help_msg)
    return parser.parse_args()

    """保存预测结果.

    :param df: 预测价格的DataFrame
    :param node_id: 节点id
    :param market_id: 日前or实时
    :return: ``None``, 结果直接写入数据库.
    """
    pt_list = tab_dict['pt_prediction_node_price']
    pt_detail = tab_dict['pt_prediction_node_price_detail']

    flag = 'da' if market_id == 3 else 'rt'
    model_id = MODEL_ID[node_id]['model_id'][flag]
    df['date'] = df.index.map(lambda x: x.date())
    df['time'] = df.index.map(lambda x: x.strftime('%H:%M'))
    df.loc[df['time'] == '00:00', 'date'] = df['date'] - datetime.timedelta(days=1)
    df.loc[df['time'] == '00:00', 'time'] = '24:00'
    start_date = df['date'].min()
    forecast_start_date = start_date - datetime.timedelta(2)
    with engine.begin() as conn:
        ret = conn.execute(select(pt_list.c.id).where(and_(
            pt_list.c.forecast_start_date == forecast_start_date.strftime('%Y-%m-%d'),
            pt_list.c.model_id == model_id
        )))
        rowcount = ret.rowcount
        if rowcount != 0:
            id_ = ret.one()[0]
            logger.info('数据库中已有结果, 删除...')
            conn.execute(delete(pt_list).where(
                pt_list.c.id == id_
            ))
            conn.execute(delete(pt_detail).where(
                pt_detail.c.prediction_node_price_id == id_
            ))

        logger.info('写入pt_prediction_node_price表...')
        ret = conn.execute(insert(pt_list), [{
            'model_id': model_id,
            'node_id': node_id,
            'market_id': market_id,
            'forecast_start_date': forecast_start_date.strftime('%Y-%m-%d'),
            'update_time': datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S')
        }])
        prediction_node_price_id = ret.inserted_primary_key[0]
        insert_values = []
        for i in range(df.shape[0]):
            insert_values.append({
                'prediction_node_price_id': prediction_node_price_id,
                'date': df.iloc[i]['date'].strftime('%Y-%m-%d'),
                'time': df.iloc[i]['time'],
                'price': post_process(df.iloc[i]['y_value']),
            })
        logger.info('写入pt_prediction_node_price_detail表...')
        conn.execute(insert(pt_detail), insert_values)
        logger.info('入库完成')

def _get_load(start_date: datetime.date, end_date: datetime.date, market_id: int) -> pd.DataFrame:
    """获取负荷类数据.

    :param start_date: 数据开始日期
    :param end_date: 数据终止日期
    :param market_id: 日前or实时
    :return: DataFrame
    """
    pt_load = tab_dict['pt_load']
    sql = select(pt_load.c.date, pt_load.c.time, pt_load.c.value, pt_load.c.power_type_id).where(
        and_(
            pt_load.c.province_id == PROVINCE_ID,
            pt_load.c.market_id == market_id,
            pt_load.c.date >= start_date.strftime('%Y-%m-%d'),
            pt_load.c.date <= end_date.strftime('%Y-%m-%d'),
            pt_load.c.power_type_id.in_([1, 2, 3])
        )
    ).order_by(pt_load.c.date, pt_load.c.time)
    df = pd.read_sql(sql, engine)
    set_datetime_col(df, False)

    # 省调负荷
    province_dispatch = df[df['power_type_id'] == 1].copy()
    province_dispatch.rename(columns={'value': 'province_dispatch_load_power'}, inplace=True)
    province_dispatch.drop(['power_type_id'], axis=1, inplace=True)
    province_dispatch.set_index('datetime', inplace=True)

    # 新能源出力
    new_energy_forecast = df[df['power_type_id'] == 2].copy()
    new_energy_forecast.rename(columns={'value': 'new_energy_load_forecast_power'}, inplace=True)
    new_energy_forecast.drop(['power_type_id'], axis=1, inplace=True)
    new_energy_forecast.set_index('datetime', inplace=True)

    # 非市场化机组出力
    not_market_gene = df[df['power_type_id'] == 3].copy()
    not_market_gene.rename(columns={'value': 'not_market_gene_power_value'}, inplace=True)
    not_market_gene.drop(['power_type_id'], axis=1, inplace=True)
    not_market_gene.set_index('datetime', inplace=True)

    df = pd.concat(
        [province_dispatch, new_energy_forecast, not_market_gene],
        axis=1)
    return df


def _get_conn(start_date: datetime.date, end_date: datetime.date, market_id: int) -> pd.DataFrame:
    """获取联络线数据.

    :param start_date: 数据开始日期
    :param end_date: 数据结束日期
    :param market_id: 日前or实时
    :return: DataFrame
    """
    pt_connection_plan = tab_dict['pt_connection_plan']
    # todo: 蒙西的联络线通道只有一条id为48的
    sql = select(pt_connection_plan.c.date, pt_connection_plan.c.time, pt_connection_plan.c.value).where(
        and_(
            pt_connection_plan.c.market_id == market_id,
            pt_connection_plan.c.province_id == PROVINCE_ID,
            pt_connection_plan.c.date >= start_date.strftime('%Y-%m-%d'),
            pt_connection_plan.c.date <= end_date.strftime('%Y-%m-%d'),
            pt_connection_plan.c.channel_id == 48

    df = pd.read_sql(sql, engine)
    df.rename(columns={'value': 'exported_power'}, inplace=True)

[docs]def set_datetime_col(df: pd.DataFrame, set_index: bool = True) -> None:
    """将date和time列合并为datetime列"""
    df['time'] = df['time'].astype(str)
    df.loc[df['time'] == '24:00', 'date'] = df['date'] + datetime.timedelta(1)
    df.loc[df['time'] == '24:00', 'time'] = '00:00'
    df['date'] = df['date'].astype(str)
    df['datetime'] = pd.to_datetime(df['date'] + ' ' + df['time'] + ':00')
    df.drop(['date', 'time'], axis=1, inplace=True)
    if set_index:
        df.set_index('datetime', inplace=True)




[docs]def set_hour(x: datetime.datetime) -> int:
    """将datetime转换为小时"""
    if x.minute == 0:
        if x.hour != 0:
            return x.hour - 1
        else:
            return 23
    else:
        return x.hour




[docs]def set_date(x: datetime.datetime) -> datetime.date:
    """将datetime转换为日期"""
    if x.hour == 0 and x.minute == 0:
        return x.date() - datetime.timedelta(days=1)
    else:
        return x.date()




[docs]def run(base_date: datetime.date, node_id: int, market_id: int = 3) -> None:
    """执行预测算法.

    :param base_date: 基准日期
    :param node_id: 节点的id
    :param market_id: 日前or实时
    :return: ``None``, 直接写入数据库.
    """
    train_start_date = base_date - datetime.timedelta(days=7)
    if market_id == 3:
        train_end_date = base_date + datetime.timedelta(days=1)
        # 只预测base_date后的第2天的价格
        pred_start_date = base_date + datetime.timedelta(days=2)
        pred_end_date = pred_start_date
    else:
        train_end_date = base_date
        # 只预测base_date后的第1天的价格
        pred_start_date = base_date + datetime.timedelta(days=1)
        pred_end_date = pred_start_date

    logger.info(f'训练数据起止日期: {train_start_date} - {train_end_date}')
    logger.info(f'预测数据起止日期: {pred_start_date} - {pred_end_date}')

    df_train = load_train_data(train_start_date, train_end_date, node_id, market_id)
    df_pred = load_pred_data(pred_start_date, pred_end_date, market_id)
    pred_index = df_pred.index

    df_train['hour'] = df_train.index.map(set_hour)
    df_train['date'] = df_train.index.map(set_date)
    df_pred['hour'] = df_pred.index.map(set_hour)
    df_pred['date'] = df_pred.index.map(set_date)

    logger.info('开始训练模型...')
    # 按日,小时求平均值
    df_train = df_train.groupby(['date', 'hour']).mean()
    df_pred = df_pred.groupby(['date', 'hour']).mean()

    model = HistGradientBoostingRegressor()
    model.fit(df_train[['province_dispatch_load_power', 'new_energy_load_forecast_power',
                        'not_market_gene_power_value', 'exported_power']], df_train['y_value'])
    logger.info('模型训练完成')

    logger.info('开始预测...')
    y_pred = model.predict(df_pred[['province_dispatch_load_power', 'new_energy_load_forecast_power',
                                    'not_market_gene_power_value', 'exported_power']])
    logger.info('预测完成')

    # 将y_pred扩展成96点
    y_pred = [x for x in y_pred for _ in range(4)]

    pred_df = pd.DataFrame({'y_value': y_pred}, index=pred_index)
    # 保留3位小数
    pred_df['y_value'] = pred_df['y_value'].map(lambda x: round(x, 3))

    logger.info('存储结果...')
    save_result(pred_df, node_id, market_id)
    logger.info('存储结果完成')
